    struct node70{

    char value;

    struct node70* left;

    struct node70* right;

    };

    void TreeInsert(node70*& root, char value){

    if (root == NULL){

    root = new node70;

    root->value = value;

    root->left = NULL;

    root->right = NULL;

    return;

    }

    if (root && root->value > value){

    TreeInsert(root->left,value);

    }

    else if (root && root->value < value){

    TreeInsert(root->right,value);

    }

    }

    void FindAllPermutations(std::vector<char> soFar, std::vector<char> rest,

    std::vector<std::vector<char> >& result)

    {

    int i,j;

    if (rest.empty()){

    result.push_back(soFar);

    }

    else{

    for (i = 0; i < rest.size(); i++){

    std::vector<char> remaining;

    for (j = 0; j < rest.size(); j++){

    if (j != i){

    remaining.push_back(rest[j]);

    }

    }

    soFar.push_back(rest);

    FindAllPermutations(soFar, remaining, result);

    soFar.pop_back();

    }

    }

    }

    bool TreeEqual(node70* x, node70* y){

    if (x && y && x->value != y->value)

    return false;

    if ( (x == NULL) ^ (y == NULL) ){

    return false;

    }

    if ( (x == NULL) && (y == NULL) ){

    return true;

    }

    if (x && y && x->value == y->value){

    return (TreeEqual(x->left,y->left) && TreeEqual(x->right,y->right));

    }

    }

    bool SetContains(std::set<node70* > zset, node70* root){

    std::set<node70* >::iterator iter;

    iter = zset.begin();

    while (iter != zset.end()){

    if (TreeEqual(*iter, root) == true){

    return true;

    }

    iter++;

    }

    return false;

    }

    void BuildTreeHelper(node70*& root, std::vector<char>& sortedvalues,

    std::set<node70* >& soFar, int j){

    std::vector<char> tmp;

    int i(0);

    int m(0);

    std::vector<char> rest;

    std::vector<std::vector<char> > result;

    std::vector<char> so;

    if (sortedvalues.size() == j){

    return;

    }

    rest = sortedvalues;

    rest.erase(rest.begin() + j);

    // find all permuations without j

    FindAllPermutations(so, rest, result);

    for (m = 0; m < result.size(); m++){

    TreeInsert(root, sortedvalues[j]);

    TreeInsert(root,result[m]);
