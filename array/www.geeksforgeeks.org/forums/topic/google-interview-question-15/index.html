    BFS normally acts as queue. But to reverse it we simply change its implementation to stack.


    
void displayBFSReverse(Node head){
    
int recursed = -1;
    
ArrayList<Node> nodeList = new ArrayList<Node>();
    
if(head == NULL) return;
    
nodeList.add(head);
    
while(recursed<nodeList.size()-1){
    
ArrayList<Node> children = new ArrayList<Node>();
    
Node temp = nodeList.get(++recursed);
    
nodeList.addAll(temp.getChildren());
    
}
    
for(int i = nodeList.size()-1; i>=0; i--){
    
System.Out.Print(nodeList.get(i).getNodeData()+"t");
    
}
    
return;
    
}
