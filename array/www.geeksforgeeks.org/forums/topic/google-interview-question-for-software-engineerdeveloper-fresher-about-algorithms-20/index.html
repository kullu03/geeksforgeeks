#include <iostream>
#include <set>

int main() {
    // Given a set of numbers [1-N] . Find the number of subsets such that the
    // sum of numbers in the subset is a prime number.

    int numbers[] = { 1,  2,  3,  4,  5};
    const int amountNumbers = sizeof(numbers)/sizeof(int);

    // Build a list of the prime numbers lesser than 100
    std::set<int> primes;
    int maxPossiblePrime = (amountNumbers * (amountNumbers+1)) / 2;
    for(int i = 2; i < maxPossiblePrime; ++i) primes.insert(i);
    for(int p = 2; p < maxPossiblePrime; ++p) {
        for (int m = 2, mult = p*m; mult < maxPossiblePrime; mult = p * (++m)) {
            primes.erase(mult);
        }
    }
    const int biggestPrime = *primes.rbegin();
    
    // use knapsack algorithm to find the elements that produce the sum until
    // the biggestPrime
    int amountOfSubset[amountNumbers+1][biggestPrime+1];
    for(int sum = 0; sum <= biggestPrime; ++sum) amountOfSubset[0][sum] = 0;
    
    for (int i = 1; i <= amountNumbers; ++i) {
        for (int w = 1; w <= biggestPrime; ++w) {
            if (numbers[i-1] <= w) {
                amountOfSubset[i][w] = std::max(1 + amountOfSubset[i-1][biggestPrime-numbers[i-1]],
                                                amountOfSubset[i-1][w]);
            } else {
                amountOfSubset[i][w] = amountOfSubset[i-1][w];
            }
        }
    }
    
    int numberSubsets = 0;
    for(std::set<int>::const_iterator itPrime = primes.begin();
                                      itPrime != primes.end();
                                      ++itPrime) {
        numberSubsets += amountOfSubset[amountNumbers][*itPrime];
    }
    std::cout << "Number of subsets that sum a prime: " << numberSubsets << std::endl;
    return 0;
}
