    Here is my solution:
    1. divide the N elements into two parts, all the odd elements go to first part.
    2. then convert the first part into (N+1)/2 consecutive integers. just simulate it on the paper, you will find out that each part is a sub-problem.
    3. more details are for you to dig out.
    (the running time complexity is o(N), and it works well while N is not larger than 1000, greater N has not been tested yet. Please correct me if anything is wrong, my pool English is included~)

    Here is my C++ code:

    /* wupimin@gmail.com
    * 17March, 2014
    */
    #include
    #include
    using namespace std;

    class IntegersArrangements
    {
    public:
    bool arrangeNIntegers(int N)
    {
    while (result.size()) result.clear();

    this->mN = N;

    arrangeRecursively(N, 1);

    /* these three irrelevant functions have be removed from this class */
    //vectorPrint();
    //return checkVector() && checkRange();

    return true;
    }
    private:
    vector result;
    int mN;

    /* 1. divide the n elements into two parts, all the odd elements go to first part.
    * just simulate it on the paper, you will find out that each part is a sub-problem.
    * more details are for you to dig out.
    * 2. then apply arrangeRecursively() on the first part until n equals 1;
    * 3. for the second part, we can arrange them similar with the first part,
    * but this time no more arrangeRecursively() is involved,
    * just add every elements in the first part with “add” ,
    * “add” is initially assigned as 1, multiply it by 2 on the next arrangeRecursively() call*/
    void arrangeRecursively(int n, int add)
    {
    int arg = add < < 1;
    if (1 == n)
    {
    result.push_back(1);
    return ;
    }

    int firstPart = ((n - 1) >> 1) + 1;
    int secondPart = n >> 1;
    arrangeRecursively(firstPart, add << 1); // apply arrangeRecursively() on the first part

    int i = 0;
    int count = 0;
    while (count < secondPart)
    {
    if (result[i] + add <= mN) // skip the elements than exceed the range of [1, mN]
    {
    result.push_back(result[i] + add); // plus every elements in the first part with “add”
    count++;
    }
    i++;
    }
    }
    };
