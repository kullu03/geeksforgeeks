    struct node * next = *head1;

    while((ref != NULL) && (next != NULL))

    next = next->next;

    prev = ref;

    ref = next;

    next = prev->next;

    }

    }

    prev->next = ref;

    }

    static private void merge(MyLinkedList list1, MyLinkedList list2) {
    MyLinkedList tmpList1 = list1;
    MyLinkedList tmpList2 = list2;

    MyNode list1Node = tmpList1.getnext();
    MyNode list2Node = tmpList2.getnext();

    while( list1Node != null && list2Node != null ) {
    MyNode tmpPrevNode = null;

    tmpPrevNode = list1Node;
    list1Node = tmpList1.getnext();

    break;
    }
    }

    tmpPrevNode.setLink(list2Node);
    }

    continue;
    }

    while( list2Node.getValue() < list1Node.getValue() ) {
    tmpPrevNode = list2Node;
    list2Node = tmpList2.getnext();
    if( list2Node == null ) {
    break;
    }
    }
    if( tmpPrevNode != null ) {
    tmpPrevNode.setLink(list1Node);
    }
    }
    }
    }
