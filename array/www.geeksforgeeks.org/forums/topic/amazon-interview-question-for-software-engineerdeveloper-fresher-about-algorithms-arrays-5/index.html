    Find least jumps to reach end of the array:

    Given an array of Integers,if u are on ith element u can make max arr jumps. If i am at an element with value zero, i cannot move forward.find the least selection to reach end of the array.

    ex: 1 3 5 8 9 2 6 7 6 8 9 starting with arr[0] i can only make one jump to 3, from 3 i can jump either 1 step 0r 2 steps 0r 3 steps.

    my solution is 1 to 3 to 8, 3 selection and i am done.

    Device an algo for this

#include <stdio.h>
    
#include <limits.h>
    

    
int minJumps(int arr[], int n)
    
{
    
int *jumps = new int[n];
    
int min;
    
jumps[n-1] = 0;
    
int i, j;
    
for (i = n-2; i >=0; i--)
    
{
    
if (arr == 0)
    
jumps     = INT_MAX;
    
else if (arr    +i >= n)
    
jumps     = 1;
    
else
    
{
    
min = INT_MAX;
    
for (j = i+1; j < n && j <= arr     + i; j++)
    
{
    
if (min > jumps[j])
    
min = jumps[j];
    
}
    
jumps     = min + 1;
    
}
    
}
    

    
return jumps[0];
    
}
    

    
int main()
    
{
    
int arr[] = {1, 3, 5, 8, 9, 2, 6, 7, 6, 8, 9};
    
int n = sizeof(arr)/sizeof(arr[0]);
    
printf("%dn", minJumps(arr, n));
    
return 0;
    
}

    You can also count jumps from left to right to solve this probem

    jumps –> Minimum number of jumps needed to reach arr from arr[0]


    
#include <stdio.h>
    
#include <limits.h>
    

    
int minJumps(int arr[], int n)
    
{
    
int *jumps = new int[n];
    
int i, j;
    

    
if (n == 0 || arr[0] == 0)
    
return INT_MAX;
    

    
jumps[0] = 0; //no need to jump at first element
    

    
// Start from the second location
    
for (i = 1; i < n; i++)
    
{
    
jumps = INT_MAX;
    
for (j = 0; j < i; j++)
    
{
    
if (i <= j + arr[j])
    
{
    
if ((jumps[j] + 1) < jumps    )
    
jumps     = jumps[j] + 1;
    
}
    
}
    
}
    
return jumps[n-1];
    
}
    

    
int main()
    
{
    
int arr[]= {1, 3, 6, 1, 0, 9};
    
int size=sizeof(arr)/sizeof(int);
    
printf("%d ",minJumps(arr,size));

}

    class MinNosOfJump
{
	int sum;
	int[] arr,path;
	MinNosOfJump(int []arr)
	{
		this.arr=arr;
		path=new int[arr.length];
		sum=0;
	}
	void getMinJump()
	{
		int startPos=1,index=0;
		int maxJumpCount=sum=arr[0];
		path[index++]=0;
		sum++;
		while(sum<=arr.length)
		{
			int max=-3;
/* 
	In case of max element is more than 1 then we should consider the 
	farthest  element. So using a decrementing for loop
*/		
			for(int i=maxJumpCount;i>=startPos&&i<arr.length;i--)
			{
				if(max<arr[i])
				{
					max=arr[i];
					startPos=i;
				}
			}
			sum+=max;
			path[index++]=startPos;
			maxJumpCount=startPos+max;
		}
	}
	public static void main(String []args)
	{
		int[] arr={5,1,1,1,1,1,4,1,1,1,2,6};
		MinNosOfJump minJump=new MinNosOfJump(arr);
		minJump.getMinJump();
		for(int i=0;i<minJump.path.length;i++)
		{
			if(i!=0)
			{	
				if(minJump.path[i]==0)
					break;
			}
			System.out.print(""+minJump.arr[minJump.path[i]]+"\t");
		}
	}
}
