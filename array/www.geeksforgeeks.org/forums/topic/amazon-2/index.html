    below is the recursive solution for Ques 2 of Round 1.
    —> I have considered root to be at level 1

    // structure of each node of a Binary Tree
    class myNode
    {
    int data;
    myNode next;
    myNode prev;
    }

    // required function
    int calSum(myNode node , int level)
    {
    int leftSum=0,rightSum=0;
    if(node.prev!=null)
    leftSum=calSum(node.prev,level+1);//calculate sum of left subtree
    else
    leftSum=0;
    if(node.next!=null)
    rightSum=calSum(node.next,level+1);//calculate sum of right subtree
    else
    rightSum=0;
    // does addition as per value of ” level ”
    if(level%2==0)
    return -node.data+(leftSum+rightSum) ;
    else
    return node.data+(leftSum+rightSum) ;
    }

    leftSum=calSum(node.prev,level+1);
    else
    leftSum=0;
    if(node.next!=null)
    rightSum=calSum(node.next,level+1);
    else
    rightSum=0;
    // doing summation as per value of “level”
    if(level%2==0)
