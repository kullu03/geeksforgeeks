    traverse preorder:-

    if stack is empty create a node and insert it in stack

    if the current element is less then the stack top , create a node with current element as data , make it the left child of stack’s top. insert this node in stack

    otherwise keep popping the elements until they are smaller then the current element. make the current element right child of the last popped element , insert it into BST

    #include <iostream>
using namespace std;

typedef struct node{

	struct node* right;
}Node;

Node* createNode(int data){
	Node* newnode = new Node;
	newnode->data = data;
	newnode->left = NULL;
	newnode->right = NULL;
	return newnode;
}

//left also includes root
//right->right starting point    
//end ->right total ending point

Node* makeTree(int* arr, int left,int right){
	Node* root = createNode(arr[left]);
	int i,l,r;
	i =left;

	while(i+1<=right && arr[i+1]<root->data){
		i++;
	}

	root->left = i==left?NULL:makeTree(arr,left+1,i);
	root->right = i==right?NULL:makeTree(arr,i+1,right);

	return root;
}

void print(Node* node){
	if(node==NULL)return;
	if(node->left!=NULL)print(node->left);
	cout<<node->data<<" ";
	if(node->right!=NULL)print(node->right);
}

int main(){
	int arr[] = {5,3,2,1,4,7,6,8};
