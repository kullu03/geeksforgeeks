    1) First Code


    
class A
    
{
    
public:
    
int i;
    
int b;
    
A(int temp){
    
i=temp;
    
A();
    
b=0;
    
}
    
A(){
    
b=0;
    
}
    
void display(){
    
printf("%d %dn",i,b);//1 0
    
}
    
};
    

    
int  main(){
    
A Aobj(1);
    
Aobj.display();
    
return 0;
    
}

    Output: 1 0

    2) Second Code


    
class A
    
{
    
public:
    
int i;
    
int b;
    
A(int temp):i(temp),A(),b(0){}//Error
    
A():b(0){}
    
void display(){
    
printf("%d %dn",b,i);
    
}
    
};
    
int  main()
    
{
    
A Aobj(1);
    
Aobj.display();
    
return 0;
    
}

    I was expecting that both the codes will show same behavior and will produce an error as calling one constructor from the other in the same class is not permitted.It’s not C++11.

    So why using the intializer list makes the difference, as I am doing the same thing in both the codes.Why the behavior is different in both the codes. I compiled this codes in g++ 4.3.4 .

    Initializer list is used to initialize members. The list specified is executed when memory is allocated to members, before the code inside constructor is executed.
