#include "stdafx.h"
    
#include "windows.h"
    

    
struct trieNode {bool endOfWord;
    
char c;
    
int count;
    
struct trieNode *children[27];
    
};
    

    
bool hasChar(struct trieNode* root, char c)
    
{
    
int index = c - 'a';
    
if (root->children[index] == 0)
    
return false;
    
else
    
return true;
    
}
    

    
void addToTrie(struct trieNode* root, char* word)
    
{
    
// we will add each character in Trie. Will increment count if it already exists
    
for (int i=0; word!= ''; i++)
    
{
    
if (!hasChar(root, word    ))
    
{
    
struct trieNode* childNode = (struct trieNode*)malloc(sizeof(struct trieNode));
    
memset(childNode, '', sizeof(struct trieNode));
    
int index = word     - 'a';
    
childNode->c = word    ;
    
childNode->count == 0;
    
root->children[index] = childNode;
    
root = childNode;
    
}
    
else
    
{
    
int index = word     - 'a';
    
struct trieNode* childNode = root->children[index];
    
root = childNode;
    
if (word[i+1] == '' && root->endOfWord)
    
{
    
root->count++;
    
}
    
}
    
}
    
root->endOfWord = true;
    
}
    

    
struct trieNode* createTrie()
    
{
    
struct trieNode* root = (struct trieNode*) malloc (sizeof(struct trieNode));
    
memset(root, '', sizeof(struct trieNode));
    
root->endOfWord = false;
    
return root;
    
}
    

    
bool searchWord(struct trieNode* root, char* word)
    
{
    
/* we will check each char of word;
    
* we will return false if any word is not found in any of the iteration and we have not reached end of word.
    
* we have reached end of word but still endOfWord is false. */
    
if (root == NULL)
    
{
    
return false;
    
}
    
int i = 0;
    
while (word     != '')
    
{
    
int index = word     - 'a';
    
if (hasChar(root, word    ))
    
{
    
root = root->children[index];
    
i++;
    
}
    
else
    
{
    
return false;
    
}
    
}
    
if (!root->endOfWord)
    
{
    
return false;
    
}
    
else
    
{
    
return true;
    
}
    
}
    

    
struct trieNode* findNextNode(struct trieNode* root, int* start)
    
{
    
int i;
    
for (i = *start; i < 27; i++)
    
{
    
if (root->children     != NULL)
    
{
    
*start = i;
    
return root->children    ;;
    
}
    
}
    
return NULL;
    
}
    

    
void printArray(char wordArray[], int length)
    
{
    
for (int i = 0; i < length; i++)
    
{
    
printf("%c", wordArray    );
    
}
    
}
    

    
int printWords(struct trieNode* root, char wordArr[], int index, int start)
    
{
    
// we pick up first non-empty slot in the root's children array and follow it to print the words
    
// we print the word when we hit endOfWord = true
    

    
// findNextNode finds first non-empty index in the array starting at position start
    
if (root == NULL)
    
{
    
return 0;
    
}
    
struct trieNode* node = findNextNode(root, &start);
    

    
if (!node)
    
{
    
return 0;
    
}
    
wordArr[index++] = start + 'a';
    

    
if (node->endOfWord)
    
{
    
printArray(wordArr, index);
    
printf(" ");
    
return start+1;
    
}
    
int nextStart;
    

    
nextStart = printWords(node, wordArr, index, 0);
    

    
if (findNextNode(node, &(nextStart)))
    
{
    
nextStart = printWords(node, wordArr, index, nextStart);
    
}
    
return start + 1;
    
}
    

    
int _tmain(int argc, _TCHAR* argv[])
    
{
    
char sentence[] = "hello amy how are you today morning";
    
char* word = NULL;
    

    
struct trieNode* trieRoot = createTrie();
    

    
word = strtok (sentence, " ");
    

    
while (word != NULL)
    
{
    
addToTrie(trieRoot, word);
    
word = strtok (NULL, " ");
    
}
    

    
char wordArr[256] = {0};
    
int nextIndex = printWords(trieRoot, wordArr, 0, 0);
    
while (nextIndex)
    
{
    
nextIndex = printWords(trieRoot, wordArr, 0, nextIndex);
    
}
    
return 0;
    
}
