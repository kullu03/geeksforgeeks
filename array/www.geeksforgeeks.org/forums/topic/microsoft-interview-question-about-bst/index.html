    Two of the nodes of a BST are swapped. Correct the BST.

    Following is a solution suggested by ICANTH 

    #include<stdio.h>
    

    
typedef struct node_targ
    
{
    
int data;
    
struct node_targ *left;
    
struct node_targ *right;
    
} Node;
    

    
typedef Node* Tree;
    

    
static inline void swap_two_nodes(Node *&node1, Node *&node2)
    
{
    
int tmp = node1->data;
    
node1->data = node2->data;
    
node2->data = tmp;
    
}
    

    
// case 1: A[1] > A[2]
    
// case 2: A[N] < A[N-1]
    
// case 3:A[i-1] > A < A[i+1]
    
// case 4:A[i-1] < A     > A[i+1]
    
static bool _adjust_two_nodes(Node *node, Node *&prev, Node *&prev_prev, Node *&first)
    
{
    
if(node == NULL)	return false;
    
else if(_adjust_two_nodes(node->left, prev, prev_prev, first))
    
{
    
return true;
    
}
    
else if(prev != NULL &&
    
(prev_prev == NULL && prev->data > node->data)
    
|| (prev_prev != NULL && prev_prev->data > prev->data && prev->data < node->data)
    
|| (prev_prev != NULL && prev_prev->data < prev->data && prev->data > node->data))
    
{ // when case1: case3: case4:
    
if(first == NULL)
    
{
    
first = prev;
    
}
    
else
    
{
    
swap_two_nodes(first, prev);
    
return true;
    
}
    
} // if
    
prev_prev = prev;
    
prev = node;
    
return _adjust_two_nodes(node->right, prev, prev_prev, first);
    
} // adjust_two_nodes
    

    
bool adjust_two_nodes(Tree &tree)
    
{
    
if(tree == NULL)	return false;
    

    
Node *prev = NULL, *prev_prev = NULL, *first = NULL;
    
if(_adjust_two_nodes(tree, prev, prev_prev, first))
    
{
    
return true;
    
}
    
else if(first != NULL)
    
{ // when case 2:
    
swap_two_nodes(first, prev);
    
return true;
    
}
    
else
    
{
    
return false;
    
}
    
} // adjust_two_nodes

    I have written following code based on method 3 of http://www.geeksforgeeks.org/archives/3042.


    
#include <stdio.h>
    
#include <stdlib.h>
    
#include <limits.h>
    
#include <stdlib.h>
    

    
/* A binary tree node has data, pointer to left child
    
and a pointer to right child */
    
struct node
    
{
    
int data;
    
struct node* left;
    
struct node* right;
    
};
    

    
void fixBSTUtil(struct node* node, int min, int max, struct node** fn);
    

    
/* Returns true if the given tree is a binary search tree
    
(efficient version). */
    
void fixBST(struct node* node)
    
{
    
struct node *fn = NULL;
    
fixBSTUtil(node, INT_MIN, INT_MAX, &fn);
    
}
    

    
void fixBSTUtil(struct node* node, int min, int max, struct node **fnr)
    
{
    
/* Base Case */
    
if (node == NULL)
    
return;
    

    
/* check if this node violates the min/max constraint */
    
if (node->data < min || node->data > max)
    
{
    
if (*fnr == NULL)
    
{
    
*fnr = node;
    
printf ("nFirst node is %d", (*fnr)->data);
    

    
// Don't change min and max values as this is a faulty node
    
fixBSTUtil(node->left, min, max, fnr);
    
fixBSTUtil(node->right, min, max, fnr);
    
}
    
else
    
{
    
printf ("nSecond node is %dn", node->data);
    
int data = (*fnr)->data;
    
(*fnr)->data = node->data;
    
node->data = data;
    
return;
    
}
    
}
    
else
    
{
    
fixBSTUtil(node->left, min, node->data-1, fnr);
    
fixBSTUtil(node->right, node->data+1, max, fnr);
    
}
    
}
    

    
/* Helper function that allocates a new node with the
    
given data and NULL left and right pointers. */
    
struct node* newNode(int data)
    
{
    
struct node* node = (struct node*)
    
malloc(sizeof(struct node));
    
node->data = data;
    
node->left = NULL;
    
node->right = NULL;
    

    
return(node);
    
}
    

    
/* Given a binary tree, print its nodes in inorder*/
    
void printInorder(struct node* node)
    
{
    
if (node == NULL)
    
return;
    

    
/* first recur on left child */
    
printInorder(node->left);
    

    
/* then print the data of node */
    
printf("%d ", node->data);
    

    
/* now recur on right child */
    
printInorder(node->right);
    
}
    

    
/* Driver program to test above functions*/
    
int main()
    
{
    
/*   4
    
/  
    
2    1
    
/ 
    
3   5
    
*/
    

    
struct node *root = newNode(4);
    
root->left        = newNode(2);
    
root->right       = newNode(1);
    
root->left->left  = newNode(3);
    
root->left->right = newNode(5);
    

    
printf("Inorder Traversal of the original tree ");
    
printInorder(root);
    

    
fixBST(root);
    

    
printf("Inorder Traversal of the fixed tree ");
    
printInorder(root);
    

    
getchar();
    
return 0;
    
}

    The code doesn’t handle the case where root is swapped.
