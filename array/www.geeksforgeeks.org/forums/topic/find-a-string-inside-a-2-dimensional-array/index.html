    	static int x[]={1,1,1,-1,-1,-1,0,0};
	static int y[]={1,0,-1,1,0,-1,1,-1};

	public static boolean findMatch(char[][] board, String str){

		boolean[][] used = new boolean[board.length][board[0].length]; 
		for(int i=0; i<board.length; i++){
			for(int j=0; j<board[0].length; j++){
				if(board[i][j] == str.charAt(0)){
					if(findneighbour(board, str, 0, i, j, used)){
						System.out.println("Found it !!!");
						return true;
					}
				}
			}
		}

		System.out.println("Cant Find it");
		return false;
	}

	public static boolean isSafe(char[][] board, int r, int c,
								 String txt, int ind, boolean[][] usedArr){
		if(r>=0 && r < board.length 
			&& c>=0 && c < board[0].length 
			&& board[r][c][/c]==txt.charAt(ind) && !usedArr[r][c][/c]){
			usedArr[r][c][/c]=true;
			return true;
		}
			
		return false;
	}

	public static boolean findneighbour(char[][] board, String str, int index,
										 int row, int col, boolean[][] usedArray){
		if(index == str.length())
			return true;

		if(isSafe(board, row, col, str, index, usedArray)){
			for(int n=0; n<8; n++){
				if (findneighbour(board, str, index+1, row+x[n], col+y[n], usedArray))
					return true;
			}
			return false;
		}
		return false;
	}

    Step-1
    find first character of the required string in the matrix and store the index in i and j
    Step-2
    find the neighbour of the index element and check with the next character of search string.
    if it is matched store the index element and recursively call it for new index until all the element of the search string have not been traversed..

    worst time complexity:O(m*n*k*8)…. k is the size of the search string.

    import copy
import numpy

def pp_mtrx(matrix):
    s = [[str(e) for e in row] for row in matrix]
    lens = [max(map(len, col)) for col in zip(*s)]
    fmt = '\t'.join('{{:{}}}'.format(x) for x in lens)
    table = [fmt.format(*row) for row in s]
    print '\n'.join(table)

class Matrix():
    rows = 0
    columns = 0
    contents = [[]]

    def __init__(self, rows, columns):
        self.rows = rows
        self.columns = columns
        self.contents = [[None for column in xrange(self.columns)] for row in xrange(self.rows)]

    def get_start_location(self, letter):
        words = list(numpy.array(self.contents).flat)

        try:
            location = words.index(letter)
        except ValueError:
            return False

        row, column = divmod(location, words.__len__())
        if column >= self.columns:
            row += divmod(column, self.columns)[0]
            column = divmod(column, self.columns)[1]

        return row, column

    def get_indexed_co_ordinates(self, location):
        co_ordinates = {'0': (location[0] - 1, location[1] - 1), '1': (location[0] - 1, location[1]), '2': (location[0] - 1, location[1] + 1),
                        '3': (location[0], location[1] - 1), '4': (location[0], location[1] + 1), '5': (location[0] + 1, location[1] - 1),
                        '6': (location[0] + 1, location[1]), '7': (location[0] + 1, location[1] + 1)
        }

        return co_ordinates

    def get_co_ordinates(self, location):
        return {'upper_left': self.get_indexed_co_ordinates(location)['0'], 'upper': self.get_indexed_co_ordinates(location)['1'],
                'upper_right': self.get_indexed_co_ordinates(location)['2'], 'left': self.get_indexed_co_ordinates(location)['3'],
                'right': self.get_indexed_co_ordinates(location)['4'], 'lower_left': self.get_indexed_co_ordinates(location)['5'],
                'lower': self.get_indexed_co_ordinates(location)['6'], 'lower_right': self.get_indexed_co_ordinates(location)['7']
                }

    def initialize_surrounding_letters(self, location):
        co_ordinates = self.get_co_ordinates(location)

        upper_left_letter = self.contents[co_ordinates['upper_left'][0]][co_ordinates['upper_left'][1]] if location[0] - 1 >= 0 and location[1] - 1 >= 0 else None
        up_letter = self.contents[co_ordinates['upper'][0]][co_ordinates['upper'][1]] if location[0] - 1 >= 0 else None
        upper_right_letter = self.contents[co_ordinates['upper_right'][0]][co_ordinates['upper_right'][1]] if location[0] - 1 >= 0 and location[1] + 1 < self.columns else None
        left_letter = self.contents[co_ordinates['left'][0]][co_ordinates['left'][1]] if location[1] - 1 >= 0 else None
        right_letter = self.contents[co_ordinates['right'][0]][co_ordinates['right'][1]] if location[1] + 1 < self.columns else None
        lower_left_letter = self.contents[co_ordinates['lower_left'][0]][co_ordinates['lower_left'][1]] if location[0] + 1 < self.rows and location[1] - 1 >= 0 else None
        lower_letter = self.contents[co_ordinates['lower'][0]][co_ordinates['lower'][1]] if location[0] + 1 < self.rows else None
        lower_right_letter = self.contents[co_ordinates['lower_right'][0]][co_ordinates['lower_right'][1]] if location[0] + 1 < self.rows and location[1] + 1 < self.columns else None

        return upper_left_letter, up_letter, upper_right_letter, left_letter, right_letter, lower_left_letter, lower_letter, lower_right_letter

    def repeat_for_next(self, locations_visited, word):
        curr_location = locations_visited[-1] if locations_visited[-1] != (None, None) else locations_visited[-2]

        try:
            found = self.search_next(locations_visited, word[1:word.__len__()])

            if found:
                return True

            self.contents[curr_location[0]][curr_location[1]] = None
            return False
        except IndexError:
            return True

    def determine_next_location(self, location, location_visited, surrounding_letters, word):
        indices = [i for i, x in enumerate(surrounding_letters) if x == word[0]]

        for i in indices:
            new_co_ordinate = self.get_indexed_co_ordinates(location)[str(i)]

            if new_co_ordinate not in location_visited:
                a = new_co_ordinate
                return a

        return None, None

    def search_next(self, location_visited, word):
        location = location_visited[-1]

        try:
            surrounding_letters = list(self.initialize_surrounding_letters(location))
        except TypeError:
            return False

        next_location = self.determine_next_location(location, location_visited, surrounding_letters, word)
        location_visited.append(next_location)

        return self.repeat_for_next(location_visited, word)

    def does_exist(self, word):
        start_letter = word[0]
        location = self.get_start_location(start_letter)

        if not location:
            return False

        found = self.search_next([location], word[1:word.__len__()])

        return True if found else self.does_exist(word)
