    For an O(log n) approach,
    1) There are two possibilities of numbers to be divisible by 5, ends with 5 or ends with 0.
    2) If it does not end with 0, then subtract 5 from it.
    3) Right shift the number by 1 bit.
    4) Repeat this until 5 is reached.

    Code:

    #include
    using namespace std;
    void check(int n)
    {
    if(n==0)
    {
    cout< <"Divisible";
    return;
    }
    while(n>5)
    {
    if(n&1==1)
    {
    n-=5;
    }
    else
    {
    n=n>>1;
    }
    }
    if(n==5)
    {
    cout< <"Divisible";
    return;
    }
    cout<<"Not Divisible";
    return;
    }
    int main()
    {
    int n;
    while(1)
    {
    cin>>n;
    check(n);
    }
    return 0;
    }

    Time Complexity: O(log n)
    Space Complexity: O(1)

    bool isDivisibleOf5(int n) {                                                       
    unsigned m = abs(n), d = 5;                                                    
    while ((d << 1) <= m) d = d << 1;                                              
    while (m >= 5) {                                                               
        if (m >= d) m = m - d;                                                     
        d = d >> 1;                                                                
    }                                                                              
    return m == 0;                                                                 
}

    S1(n); //input
    for(i=0;p10[i]<=n&&i<19;i++); // O(log n)
    i--;
