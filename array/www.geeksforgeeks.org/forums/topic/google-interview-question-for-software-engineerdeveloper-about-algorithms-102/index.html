    You are given a String number containing the digits of a phone number

    (the number of digits, n, can be any positive integer) . To help you memorize the number, you want to divide it into groups of contiguous digits. Each group must contain exactly 2 or 3 digits. There are three kinds of groups:

    • Excellent: A group that contains only the same digits. For example, 000 or 77.

    • Good: A group of 3 digits, 2 of which are the same. For example, 030, 229 or 166.

    • Usual: A group in which all the digits are distinct. For example, 123 or 90.

    The quality of a group assignment is defined as 2 × (number of excellent groups) + (number of good groups)

    Divide the number into groups such that the quality is maximized.

    Design an efficient algorithm to return the solution that maximizes the quality.

    package com.prepare;
    

    
import java.util.ArrayList;
    
import java.util.Arrays;
    
import java.util.List;

    /**

    * You are given a String number containing the digits of a phone number (the

    * number of digits, n, can be any positive integer) . To help you memorize the

    * number, you want to divide it into groups of contiguous digits. Each group

    * must contain exactly 2 or 3 digits. There are three kinds of groups: •

    * Excellent: A group that contains only the same digits. For example, 000 or

    * 77. • Good: A group of 3 digits, 2 of which are the same. For example, 030,

    * 229 or 166. • Usual: A group in which all the digits are distinct. For

    * example, 123 or 90. The quality of a group assignment is defined as 2 ×

    * (number of excellent groups) + (number of good groups) Divide the number into

    * groups such that the quality is maximized. Design an efficient algorithm to

    * return the solution that maximizes the quality.

    * http://geeksforgeeks.org/forum/topic/google-interview-question-for-software-engineerdeveloper-about-algorithms-76

    *

    * @author hrbapna

    *

    */


    
public class GroupOfContinousDigits {
    

    
// String phoneNumber ;
    
int[] phnNumDigits;
    

    
// Depending on question requirement we can create string or int array
    
// group.
    
List<String> excellentGrp = new ArrayList<String>();
    
List<String> goodGrp = new ArrayList<String>();
    
List<String> usualGrp = new ArrayList<String>();
    

    
int quality;
    

    
public GroupOfContinousDigits(String phn) {
    

    
phnNumDigits = new int[phn.length()];
    
char[] asChar = phn.toCharArray();
    
int i = 0;
    
for (char c : asChar) {
    
phnNumDigits[i++] = Character.digit(c, 10);
    
}
    
Arrays.sort(phnNumDigits);
    

    
}
    

    
public static void main(String[] args) {
    
String phn = "9829146200";
    
GroupOfContinousDigits grpOfContDig = new GroupOfContinousDigits(phn);
    
grpOfContDig.divdIn2GrpsThtMaxmizeQuality();
    
int quality = grpOfContDig.calcQuality();
    
System.out.println("Quality: " + quality);
    

    
System.out.println("Exc: " + grpOfContDig.excellentGrp);
    
System.out.println("Gud: " + grpOfContDig.goodGrp);
    
System.out.println("Usu: " + grpOfContDig.usualGrp);
    
}
    

    
private int calcQuality() {
    
quality = 2 * excellentGrp.size() + goodGrp.size();
    
return quality;
    
}
    

    
private void divdIn2GrpsThtMaxmizeQuality() {
    

    
int length = phnNumDigits.length;
    
int i = 0;
    
while (i < length - 1) {
    

    
if (phnNumDigits == phnNumDigits) {
    
excellentGrp.add("" + phnNumDigits     + phnNumDigits);
    
i = i + 2;
    
} else if (i + 2 < length) {
    
if (phnNumDigits     == phnNumDigits) {
    
swap(i, i + 2);
    
excellentGrp.add("" + phnNumDigits     + phnNumDigits);
    
i = i + 2;
    
} else {
    
usualGrp.add("" + phnNumDigits     + phnNumDigits);
    
i = i + 2;
    
}
    
}
    
}
    

    
}
    

    
private void swap(int l1, int l2) {
    
int temp = phnNumDigits[l1];
    
phnNumDigits[l1] = phnNumDigits[l2];
    
phnNumDigits[l2] = temp;
    
}
    
}

    here my java solution (it works, I’ve tested it):


    

    
package it.coding.ex;
    

    
import java.util.ArrayList;
    
import java.util.HashSet;
    
import java.util.List;
    

    
public class BestGrouping {
    

    
private static String bestGroups = "";
    
private static int bestQuality = 0;
    

    
public static void updateBestGrouping(List<List<String>> groups) {
    
int quality = 0;
    
int excellent = 0;
    
int good = 0;
    
for (List<String> group : groups) {
    
HashSet<String> set = new HashSet<String>();
    
for (String digit : group) {
    
set.add(digit);
    
}
    
if (set.size() == 1) excellent++;
    
if (set.size() == 2 && group.size() == 3) good++;
    
}
    

    
quality = (2 * excellent) + good;
    
System.out.println(quality + " => " + groups);
    

    
if (quality >= bestQuality) {
    
bestQuality = quality;
    
bestGroups = "" + groups; // deep copy of the list
    
}
    
}
    

    
public static void bestGrouping(List<String> number, List<List<String>> groups) {
    
if (number.isEmpty()) {
    
updateBestGrouping(groups);
    
}
    
else {
    
if (number.size() < 2) return;
    

    
List<String> group = new ArrayList<String>();
    
group.add(number.get(0));
    
group.add(number.get(1));
    

    
groups.add(group);
    
bestGrouping(number.subList(2, number.size()), groups);
    

    
if (number.size() > 2) {
    
group.add(number.get(2));
    
bestGrouping(number.subList(3, number.size()), groups);
    
}
    

    
groups.remove(groups.size() - 1);
    
}
    
}
    

    
public static void main(String[] args) {
    

    
List<String> number = new ArrayList<String>();
    
String numberStr = "123354401099";
    
for (int i = 0; i < numberStr.length(); i++) {
    
number.add("" + numberStr.charAt(i));
    
}
    

    
bestGrouping(number, new ArrayList<List<String>>());
    
System.out.println("Best quality of " + bestQuality + " for group " + bestGroups);
    
}
    

    
}
