    while doing level-order traversal, you can compute the size of the next level:

    
int BinaryTree::width()
{
	queue<Node*> q;
	if(!root)
		return 0;
	q.push(root);
	int i = 1;
	int next_level_size= 0;
	int max = 1;
	do
	{
		while(i>0)
		{
			Node * p = q.front();
			q.pop();
			i--;

			if(p->left)
			{
				q.push(p->left);
				next_level_size++;
			}
			if(p->right)
			{
				q.push(p->right);
				next_level_size++;
			}
		}
		if(max < next_level_size)
			max = next_level_size;
		i = next_level_size;
		next_level_size = 0;
	}while(i>0);
	return max;
}
