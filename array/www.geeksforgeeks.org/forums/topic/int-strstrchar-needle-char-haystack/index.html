    char *my_strstr(const char *haystack, const char *needle)
    {
    int i = 0, j = 0;
    /* loop till end of src string */
    while(haystack[i]) {
    if( haystack[i] == needle[j])
    j++; /* increment the needle if char matches */
    else
    j = 0; /* reset the needle */
    if( !needle[j]) /* check for the end of string and return the addrs of haystack */
    return (haystack + i -j + 1);
    i++;
    }
    return (char)’\0′; /* NULL if pattern not found */
    }
