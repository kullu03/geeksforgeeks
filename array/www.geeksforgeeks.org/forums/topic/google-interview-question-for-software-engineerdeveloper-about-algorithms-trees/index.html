    By inorder traversal, store all elements in an array. Now the problem of

    finding out max size BST is reduced to finding out max sorted

    contiguous sub sequence. Below is the method to find out that.

    int maxSorted(int a[], int n)
    
{
    
int s[n] = {0,};    //Dynamic arrays are feature of C99
    
int i;
    

    
for(i = n - 2;  i  >= 0; --i)
    
{
    
if(a < a)
    
s     = s[i+1] + 1;
    
}
    

    
int max = s[n-1];
    
for( i = n -2; i >= 0; --i)
    
{
    
if(a     > max)
    
max = a    ;
    
}
    
return max;
    
}
