map<char, int> &GetMap(void) {
    

    
static map<char, int> m;
    

    
m =  1000;
    
m =  500;
    
m =  100;
    
m =  50;
    
m =  10;
    
m =  5;

long RomanToDecimal(string s) {
    
long result = 0;
    
int prev = 0;
    
int digit;
    

    
map<char, int> &m = GetMap();
    

    
for(size_t i = 0; i < s.size(); i++) {
    
digit = m[s];
    
result += digit;
    
if( prev < digit )
    
result -= (2*prev); // last digit already added, so subtract twice
    
prev = digit;
    
}
    

    
return result;
    
}
