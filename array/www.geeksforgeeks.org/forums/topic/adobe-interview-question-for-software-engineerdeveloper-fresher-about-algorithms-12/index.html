    inline bool isBitSet(unsigned int var, unsigned int pos)
    
{
    
return ((var & (1 << (pos))) ? true : false);

bool isPalindrome(unsigned int var)
    
{
    
bool isPalin = true;
    
unsigned int pos  = 0;
    

    
if(var == 0)
    
{
    
return isPalin;
    
}
    

    
// Check the position of the most significant bit
    
for(int i = sizeof(var)*8; i >= 0; i--)
    
{
    
if(isBitSet(var, i))
    
{
    
pos = (unsigned int)i;
    
break;
    
}
    
}
    

    
// Now compare the corresponding bits
    
// i.e. if the input variable has 5 bits (Non zero bit set at 5th bit) then
    
// compare 5th with 0th, then 4th with 1st...
    
for(unsigned int i = 0; i < (pos + 1)/2; i++)
    
{
    
if(isBitSet(var, pos) != isBitSet(var, pos-i))
    
{
    
isPalin = false;
    
break;
    
}
    
}
    

    
return isPalin;
    
}

// Returns if pos'th bit (in 0 based indexing) is 1
bool isBitSet(int n, int pos) {
	return n & (1 << pos);
}

bool isPalindrome(int n) {
	int noOfBits = sizeof(n) * 8;
	
	for (int i = 0; i < noOfBits / 2; i++) {
		// Check if bits at position i and position (noOfBits - i - 1) are equal.
		// e.g. for a 8 bit number, it will first check for 0'th and 7'th bit,
		// 1'th and 6'th bit and so on
		if (isBitSet(n, i) != isBitSet(n, noOfBits - i - 1)) {
			return false;
		}
	}
	
	return true;
}
