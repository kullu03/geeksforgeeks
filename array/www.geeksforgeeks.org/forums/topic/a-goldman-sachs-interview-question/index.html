    void remove_spaces(char c[], int n)

    {

    int i=0,j=-1;

    while(i < n)

    {

    if(c==’ ‘ && j == -1)

    j = i;

    else if(c != ‘ ‘ && j != -1)

    {

    char temp = c;

    c = c[j];

    c[j] = temp;

    j++;

    }

    i++;

    }

    }
