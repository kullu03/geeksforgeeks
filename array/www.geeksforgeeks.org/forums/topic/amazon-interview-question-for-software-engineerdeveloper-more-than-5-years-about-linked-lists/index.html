    Here is a proposed solution that take O(N) time best case and O(N log N) time on randomly created singly linked list. FrankC

    struct SLLNode{
    
int value;
    
struct SLLNode* next;
    
struct SLLNode* nextLargest;
    
};
    

    
struct DLLNode{
    
int value;
    
struct DLLNode* prev;
    
struct DLLNode* next;
    
struct DLLNode* nextLargest;
    
struct DLLNode* parent;
    
struct SLLNode* originalNode;
    
};
    

    
DLLNode* FindAncestor(struct DLLNode* current){
    
DLLNode* tmp;
    
if (current == NULL){
    
return NULL;
    
}
    
tmp = current;
    
while(tmp->parent){
    
tmp = tmp->parent;
    
}
    
return tmp;
    
}
    

    
void ConvertSLLToDLL(struct DLLNode*& root, struct DLLNode* parent, struct SLLNode* current){
    
if (current == NULL){
    
return;
    
}
    
if (root == NULL){
    
root = new DLLNode;
    
root->value = current->value;
    
root->prev = NULL;
    
root->next = NULL;
    
root->nextLargest = NULL;
    
root->originalNode = current;
    
root->parent = parent;
    
}
    
else if (root->value < current->value){
    
ConvertSLLToDLL(root->next, root, current);
    
}
    
else if (root->value > current->value){
    
ConvertSLLToDLL(root->prev, root, current);
    
}
    
}
    

    
void ConvertDLLNodeToSLLNode(struct DLLNode* root){
    
DLLNode* ancestor;
    
if (root == NULL)
    
return;
    
ConvertDLLNodeToSLLNode(root->prev);
    
if (root->parent && !root->next && root->parent->value > root->value){
    
root->originalNode->nextLargest = root->parent->originalNode;
    
}
    
if (root->parent && !root->next && root->parent->value < root->value){
    
ancestor = FindAncestor(root->parent);
    
if (ancestor){
    
root->originalNode->nextLargest = ancestor->originalNode;
    
}
    
}
    
else if (root->next){
    
root->originalNode->nextLargest = root->next->originalNode;
    
}
    
ConvertDLLNodeToSLLNode(root->next);
    
}

    Now for each node value in linked list, search the next greater value in array using binary search which takes logn time.

    Here the original linked list is not affected i.e. its an in place algorithm.

    void ConvertDLLNodeToSLLNode(struct DLLNode* root){

    DLLNode* ancestor;

    if (root == NULL)

    return;

    ConvertDLLNodeToSLLNode(root->prev); // By Induction

    // Set the root->originalNode->nextLargest pointer to correct value

    // root->originalNode is a single linked list node

    if (root->parent && !root->next && root->parent->value > root->value){

    root->originalNode->nextLargest = root->parent->originalNode;

    }

    if (root->parent && !root->next && root->parent->value < root->value){

    // Determine the parent of root->parent if one exists

    ancestor = FindAncestor(root->parent);

    if (ancestor){

    root->originalNode->nextLargest = ancestor->originalNode;

    }

    }

    else if (root->next){ // self explanatory

    root->originalNode->nextLargest = root->next->originalNode;

    }

    ConvertDLLNodeToSLLNode(root->next); // By Induction

    }

    struct node* SortedMerge(struct node* a, struct node* b)

    return(b);

    push the first number to a stack. 
  for every number, 
     while stack is not empty
      check if the number is greater than the top of the stack, 
         if true, 
           then for the element on the top of the stack, the current number is the next larger. 
           remove the top element from the stack.
         else
           break (while stack is not empty)
     add the current number to the top of the stack

    Running time is o(n) (at max two iterations per number)
    Space is o(n) (at max n elements on stack).
