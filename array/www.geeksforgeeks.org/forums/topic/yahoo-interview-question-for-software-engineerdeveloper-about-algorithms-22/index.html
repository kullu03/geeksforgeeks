// tail.cpp : Defines the entry point for the console application.
    
//
    

    
#include "stdafx.h"
    
#include <iostream>
    
#include <string>
    
#include <queue>
    
#include <fstream>
    
#include <iterator>
    

    
using namespace std;
    

    
class line_item
    
{
    
private:
    
int lineNumber;
    
string lineContent;
    

    
public:
    
line_item(int line, string content)
    
{
    
lineNumber = line;
    
lineContent = content;
    
}
    

    
int get_line_number()
    
{
    
return lineNumber;
    
}
    

    
string get_line_content()
    
{
    
return lineContent;
    
}
    

    
};
    

    
int _tmain(int argc, _TCHAR* argv[])
    
{
    
deque<line_item> queue;
    

    
ifstream in;
    
int lineNumber = 0;
    
size_t lineCount = 20;
    

    
if (argc != 4)
    
{
    
cout << "Usage: tail -n [count] file" << endl;
    
return -1;
    
}
    

    
lineCount = _wtoi(argv[2]);
    

    
in.open(argv[3]);
    

    
if (in.is_open())
    
{
    
while (in.good())
    
{
    
string readLine;
    
getline(in, readLine);
    
queue.push_front( line_item(++lineNumber, readLine) );
    
if (queue.size() > lineCount)
    
{
    
queue.pop_back();
    
}
    
}
    

    
while (!queue.empty())
    
{
    
cout << "[" << queue.back().get_line_number() << "] "
    
<< queue.back().get_line_content() << endl;
    
queue.pop_back();
    
}
    

    
in.close();
    
}
    

    
return 0;
    
}

if (ch=='n')
    
display(fp,N-1);
    
else

    @Ankit:

    -2 is there because, in some editors when you finish writing and save your document, it automatically appends a new line at the end of the file. If we use -1 instead of -2, we would end up reading n-1 lines instead of n lines (since it’ll count the last ‘n’ as a line).

    In above algorithm, I directly jumped to the end of the file and from there I start reading the file in reverse direction and started printing the characters in bottom-up fashion in the recursive way.

    “Else” portion is required because we want only one of the recursive statement to be executed. Instead of “else” portion you can use another “if” statement, that will obviously serve the purpose. You just need to make sure only one (and appropriate) recursive call is made.
