    static void MaxMinInArray(int[] _arr)

    {

    if (_arr == null) return;

    int min =_arr[0];

    int max =_arr[0];

    for (int i = 0; i <= _arr.GetUpperBound(0); i += 2)

    {

    int tempMin;

    int tempMax;

    if (!((i + 1) > _arr.GetUpperBound(0)))

    {

    tempMin = Math.Min(_arr, _arr);

    tempMax = Math.Max(_arr, _arr);

    }

    else

    {

    tempMin = _arr;

    tempMax =_arr;

    }

    if (tempMax > max)

    {

    max = tempMax;

    }

    if (tempMin < min)

    {

    min = tempMin;

    }

    }
