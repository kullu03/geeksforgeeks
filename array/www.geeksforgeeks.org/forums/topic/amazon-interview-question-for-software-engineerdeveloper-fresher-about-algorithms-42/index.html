    right-tree->7

    Call recursively.

    A Java implementation based on @ila solution.

    public TreeNode buildBST(List<Integer> a) {

    if (a.size() == 0)

    return null;

    TreeNode root = new TreeNode(a.get(0));

    List<Integer> leftSubTree = new ArrayList<Integer>();

    List<Integer> rightSubTree = new ArrayList<Integer>();

    for (Integer i : a) {

    if (i > a.get(0))

    rightSubTree.add(i);

    else if(i < a.get(0))

    leftSubTree.add(i);

    // else must exclude the equals condition

    }

    root.left = buildBST(leftSubTree);

    root.right = buildBST(rightSubTree);

    return root;

    }
