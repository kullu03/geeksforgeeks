    void shellsort(int a[],int array_size) {
    
int i,j,incr,temp;
    
incr=3;
    
while(incr>0)
    
{
    
for(i=0;i<array_size;i++) {
    
j=i;
    
temp=a;
    
while ( (j >=incr)&&(a[j-incr] >temp)) {
    
a[j]=a[j-incr];
    
j=j-incr;
    
}
    
a[j] = temp;
    
}
    
if (incr/2 !=0)
    
incr = incr/2;
    
else if (incr == 1)
    
incr =0;
    
else
    
incr=1;
    
}
    
}
