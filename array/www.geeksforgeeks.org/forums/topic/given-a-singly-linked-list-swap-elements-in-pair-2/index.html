/* Function to alternatively swap the elements */
    
void rearrange(struct node *start)
    
{
    
struct node *p, * q;
    
int temp;
    
if ((!start) || !start->next)
    
return;
    
p = start;
    
q = start->next;
    
while(q)
    
{
    
temp = p->data;
    
p->data = q->data;
    
q->data = temp;
    
p = q->next;
    
q = p?p->next:0;
    
}
    
}
