    int removeDuplicates(int A[], int n) {
    
if(n<2)return n;
    
int i=1, j=1;
    
while(i<n)
    
{
    
if(A[i-1]==A)i++;
    
else A[j++]=A[i++];
    
}
    
return j;
    
}
