class queue{
	private:
		int front;
		int rear;
		int maxsize;
		vector<bdptr> qvect;
	public:
		queue():maxsize(50),front(-1),rear(-1)
		{
			qvect.reserve(maxsize);
		}
		queue(int n):maxsize(n),front(-1),rear(-1)
		{
			qvect.reserve(maxsize);
		}
		bool isempty()
		{
			if(front==-1)
			return true;
			else
			return false;
		}
		bool isfull()
		{
			if((rear+1)%maxsize==front)
			return true;
			else
			return false;
		}
		void enqueue(bdptr k)
		{
			if(isfull())
			cout<<"\nQueue is full!!\n\n";
			else
			{
				if(isempty())
				{
					front++;
					rear++;
					qvect[front]=k;
				}
				else
				{
					qvect[(++rear)%maxsize]=k;
				}
			}
		}
		bdptr dequeue()
		{
			if(isempty())
			cout<<"\nQueue is empty!!\n\n";
			else
			{
				if(rear==front)
				{
					bdptr t=qvect[front];
					front=-1;
					rear=-1;
					return t;	
				}
				else
				{
					bdptr t=qvect[front];
					front=(front+1)%maxsize;
					return t;
				}
			}
		}
};
