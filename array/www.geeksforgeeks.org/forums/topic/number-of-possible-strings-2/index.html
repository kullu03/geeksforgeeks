    There is a string of n characters. It can be made of only a or b or c., with the following restrictions:

    1. No restriction on a

    2. b can occur at most once in the whole string

    3. c can occur simultaneously only once, And the maximum length of it occurring simultaneously can be only 2. it can occur independently any number of times.

    Example: aaaaaaaaaa, acacacacac, aaabccaaca are valid strings of size 10

    aaaabbbbcc is invalid because b occurs twice

    aaccc is invalid because c is occurring simultaneously with length > 2

    aaccaabcc is invalid because c is occurrimg simultaneously twice .

    Find the number of strings possible.

    Here is the recursive approach based on backtracking:

    void countString(char *str,int depth,int N,int *count)
    
{
    
static int setB,setC,i;
    
if(N<1)
    
return;
    
if(depth==N)
    
{
    
(*count)++;
    
for(i=0;i<N;i++)
    
printf("%c",str);
    
printf("n");
    
return;
    
}
    
else
    
{
    
str[depth]='a';
    
countString(str,depth+1,N,count);
    

    
if(!setB)
    
{
    
str[depth]='b';
    
setB=1;
    
countString(str,depth+1,N,count);
    
setB=0;
    
}
    
if(!setC)
    
{
    
if(str[depth-1]=='c')
    
{
    
str[depth]='c';
    
setC=1;
    
countString(str,depth+1,N,count);
    
setC=0;
    
}
    
else
    
{
    
str[depth]='c';
    
countString(str,depth+1,N,count);
    
}
    
}
    
}
    
}
    

    
int main()
    
{
    
int count=0,depth=0,N;
    
char str[10];
    
printf("Enter N ");
    
scanf("%d",&N);
    

    
countString(str,depth,N,&count);
    
printf("Number of strings is : %dn",count);
    

    
return 0;
    

    
}

    This code solves the problem faster. 

    #include "stdio.h"
#define MAX 10000

static int anbnc2[MAX],abnc2[MAX],abc2[MAX],anbc2[MAX],bnc2[MAX],bc2[MAX],cnbnc2[MAX],cbnc2[MAX],cnbc2[MAX],cbc2[MAX],answer[MAX];

int func(int x){
	if (answer[x] != 0){
		return answer[x];
	}

	anbnc2[x]	= anbnc2[x-1] 	+ cnbnc2[x-1];
	abnc2[x] 	= abnc2[x-1] 	+ cbnc2[x-1]	+ bnc2[x-1];
	abc2[x]  	= abc2[x-1] 	+ bc2[x-1]		+ cbc2[x-1];
	anbc2[x] 	= anbc2[x-1] 	+ cnbc2[x-1];

	bnc2[x]  	= anbnc2[x-1]	+ cnbnc2[x-1];
	bc2[x]	 	= anbc2[x-1]	+ cnbc2[x-1];

	cnbnc2[x]	= anbnc2[x-1];
	cbnc2[x] 	= abnc2[x-1] 	+ bnc2[x-1];
	cbc2[x]  	= abc2[x-1] 	+ bc2[x-1]		+cbnc2[x-1];
	cnbc2[x] 	= anbc2[x-1] 	+ cnbnc2[x-1];

	answer[x]  = anbnc2[x]+abnc2[x]+abc2[x]+anbc2[x]+bnc2[x]+bc2[x]+cnbnc2[x]+cbnc2[x]+cbc2[x]+cnbc2[x];
	return answer[x];
}

    There is a dp solution. we can maintain a dp array like dp[position][ending with char a or b or c][bool for occurence of ‘b’][bool for occurence of ‘cc’]. Then the code is simple. Here is the pseudo code

    dp[i][a][0][0] = dp[i – 1][a][0][0] + dp[i – 1][c][/c][0][0];
    dp[i][a][0][1] = dp[i – 1][a][0][1] + dp[i – 1][c][/c][0][1];
    dp[i][a][1][0] = dp[i – 1][a][1][0] + dp[i – 1][c][/c][1][0] + dp[i – 1][b][1][0];
    dp[i][a][1][1] = dp[i – 1][a][1][1] + dp[i – 1][c][/c][1][1] + dp[i – 1][b][1][1];

    dp[i][b][1][0] = dp[i – 1][a][0][0] + dp[i – 1][c][/c][0][0];
    dp[i][b][1][1] = dp[i – 1][a][0][1] + dp[i – 1][c][/c][0][1];

    dp[i][c][/c][0][0] = dp[i – 1][a][0][0];
    dp[i][c][/c][0][1] = dp[i – 1][a][0][1] + dp[i – 1][c][/c][0][0];
    dp[i][c][/c][1][0] = dp[i – 1][a][1][0] + dp[i – 1][b][1][0];
    dp[i][c][/c][1][1] = dp[i – 1][a][1][1] + dp[i – 1][c][/c][1][0] + dp[i – 1][b][1][1];
