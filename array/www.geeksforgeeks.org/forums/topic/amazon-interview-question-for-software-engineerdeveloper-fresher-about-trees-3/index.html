void CompleteTree(node** root)
    
{
    
if(!root) return;
    

    
queue<node* > q;
    
q.push(*root);
    

    
while(!q.empty())
    
{
    
node* n1 = q.front();
    
if(n1->left) q.push(n1->left);
    
if(n1->right) q.push(n1->right);
    
q.pop();
    

    
node* n2 = NULL;
    
if(!q.empty())
    
{
    
n2 = q.front();
    
}
    
if(n2 != NULL && n2->data > n1->data)
    
{
    
n1->sibling = n2;
    
}
    
else
    
{
    
n1->sibling = NULL;
    
}
    
}
    
}

    I was asked the same question in microsoft interview. I gave the answer similar to the approach of a level order traversal. Then they asked me to think t=of a solution without using any data structure.

    rajcools approach is correct….. my approach is similar but implementation is different.


    
struct tree_ext
    
{
    
int data;
    
tree_ext* left;
    
tree_ext* right;
    
tree_ext* next;
    
};
    

    
typedef tree_ext* Tree_ext;
    

    
void LevelConnect(Tree_ext root)
    
{
    
if (!root)
    
return;
    

    
tree_ext* ptr = root;
    
tree_ext* child_prev = NULL;
    
while (ptr)
    
{
    
if (ptr->left)
    
{
    
if (!child_prev)
    
{
    
child_prev = ptr->left;
    
}
    
else
    
{
    
child_prev->next = ptr->left;
    
child_prev = ptr->left;
    
}
    
}
    
if (ptr->right)
    
{
    
if (!child_prev)
    
{
    
child_prev = ptr->right;
    
}
    
else
    
{
    
child_prev->next = ptr->right;
    
child_prev = ptr->right;
    
}
    
}
    
ptr = ptr->next;
    
}
    

    
LevelConnect(root->left);
    
LevelConnect(root->right);
    
}
