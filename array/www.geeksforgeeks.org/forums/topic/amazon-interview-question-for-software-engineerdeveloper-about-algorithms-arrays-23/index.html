    This is my java code for this problem. If anyone can find any counter example of this solution, then let me know.

    public static int getMaxDiff(int[] arr) {
    
if (arr == null) {
    
return -1;
    
}
    

    
Stack<Integer> posStack = new Stack<Integer>();
    
Stack<Integer> negStack = new Stack<Integer>();
    
int maxDiff = -1;
    
for (int i = 0; i < arr.length; i++) {
    
if (arr < 0) {
    
if (!posStack.isEmpty()) {
    
if (posStack.peek() - arr     > maxDiff) {
    
maxDiff = posStack.peek() - arr    ;
    
}
    
}
    
if (negStack.isEmpty() || negStack.peek() > arr    ) {
    
negStack.push(arr    );
    
}
    
} else {
    
if (!negStack.isEmpty()) {
    
if (arr     - negStack.peek() > maxDiff) {
    
maxDiff = arr     - negStack.peek();
    
}
    
}
    
if (posStack.isEmpty() || arr     > posStack.peek()) {
    
posStack.push(arr    );
    
}
    
}
    
}
    
return maxDiff;
    
}

    private static int findMaxDiff(int[] arr) {
  int minIndex=0,maxIndex=0,tempMinIndex=0;
  for ( int i=1 ; i<arr.length ; i++ ){
    if ( arr[i] > arr[maxIndex] || arr[i]-arr[tempMinIndex] > arr[maxIndex]-arr[minIndex]){
      maxIndex=i;
      minIndex=tempMinIndex;
    } else if ( arr[i]<arr[tempMinIndex]){
      tempMinIndex=i;
    }
  }
  return arr[maxIndex]-arr[minIndex];
}

    public static int maxDiff(int[] inputs){

    int diff, max = 0;//keep positive

    for(int i=0; i<inputs.length;i++){

    for(int j=0; j<i; j++){

    diff = inputs[i] – inputs[j];

    if(diff > max){

    max = diff;

    }

    }

    }

    return max;

    }
