    #include<iostream>

    #include<cstring>

    using namespace std;

    void reverse(char str[])

    { int len=strlen(str),i;

    for(i=0;i<len/2;i++)

    {

    str=str^str[len-i-1];

    str[len-1-i]=str^str[len-i-1];

    str=str^str[len-1-i];

    }

    };

    int main()

    { char str[400];

    cin>>str;

    reverse(str);

    cout<<str;

    return 0;

    }
