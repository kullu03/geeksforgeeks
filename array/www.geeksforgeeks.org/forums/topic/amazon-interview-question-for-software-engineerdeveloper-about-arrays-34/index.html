    My approach – works in O(n) time and O(1) space ..
    I tested it thoroughly .. Do comment if you find it failing ..

    #include
    using namespace std;
    void getDuplicates(int a[], int size);
    void disp(int a[], int size);

    int main(int argc, char const *argv[])
    {
    int m=3, n=5;
    int size = m+n+1;
    // int a[] = {-1, 1, 5, 3, 4, 3, 5, 2, 2}; // array starts from index 1
    // int a[] = {-1, 1, 2, 3, 4, 5, 1, 1, 2};
    // int a[] = {-1, 1, 2, 3, 4, 5, 1, 1, 1};
    int a[] = {-1, 1, 1, 1, 1, 1, 1, 1, 1};

    getDuplicates(a, size);

    return 0;
    }

    void getDuplicates(int a[], int size)
    {
    int i, t, x;

    for(i=1; i if(a[i] == i)
    a[i] = -1;
    else {
    t = a[i];
    while(t>0) {
    x = a[t];
    if(a[t] > 0) a[t] = -1;
    else a[t] -= 1;
    t = x;
    }
    }
    }

    disp(a, size);

    for(i=1; i if(a[i] < -1)
    cout << i << ” “;
    cout << “\n”;
    }

    void disp(int a[], int size)
    {
    int i;

    for(i=1; i cout << a[i] << ” “;
    cout << “\n”;
    }
