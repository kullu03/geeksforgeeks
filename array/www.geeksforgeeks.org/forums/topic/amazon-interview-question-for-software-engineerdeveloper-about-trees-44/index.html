    import java.util.ArrayList;

    import junit.framework.TestCase;

    public class IdenticalBST extends TestCase {

    /**

    * 1. Check if stream is finished if yes -> return false

    * 2. Check if both the stream has same number of elements, if not -> return false

    ? * 3. Check if root is same, if not -> return false

    ? * 4. leftsubtree arrays = constructed from smaller elements than root

    ? * 5. rightsubtree arrays = constructed from greater elements than root

    ?* 6. if leftsubtree == rightsubtree -> return true

    ?* 7. For finding result in step 6, follow above steps recursively

    * @param a

    * @param b

    * @return

    */

    public boolean isIdenticalBST(Integer[] a, Integer[] b) {

    if (a == null || a.length < 1 || b == null || b.length < 1)

    return false;

    if (a.length != b.length)

    return false;

    if (!a[0].equals(b[0]))

    return false;

    if (a.length == 1)

    return true;

    List<Integer> a_left_sub_tree = new ArrayList<Integer>();

    List<Integer> a_right_sub_tree = new ArrayList<Integer>();

    List<Integer> b_left_sub_tree = new ArrayList<Integer>();

    List<Integer> b_right_sub_tree = new ArrayList<Integer>();

    for (int i : a) {

    if (i < a[0])

    a_left_sub_tree.add(i);

    else if (i > a[0])

    a_right_sub_tree.add(i);

    // else ???????

    }

    for (int j : b) {

    if (j < b[0])

    b_left_sub_tree.add(j);

    else if (j > b[0])

    b_right_sub_tree.add(j);

    // else ???????

    }

    // ????a?b?????????

    return isIdenticalBST(a_left_sub_tree.toArray(new Integer[] {}),

    b_left_sub_tree.toArray(new Integer[] {}))

    b_right_sub_tree.toArray(new Integer[] {}));

    }

    public void testIsIdenticalBST() throws Exception {

    Integer[] a = { 10, 5, 20, 15, 30 };

    Do inorder traversal at the same time using two stacks while the elements popped up are of same value

    The tress are identical if the both stacks are empty

    public boolean compare(TreeNode root1, TreeNode root2)

    {

    Stack<TreeNode> s1 = new Stack<TreeNode>();

    Stack<TreeNode> s2 = new Stack<TreeNode>();

    for(TreeNode node = root1; node != null; node = node.left) s1.push(node);

    for(TreeNode node = root2; node != null; node = node.left) s2.push(node);

    while(!s1.empty() && !s2.empty() && s1.peek().val == s2.peek().val) {

    TreeNode node1 = s1.pop();

    for(TreeNode node = node1.right; node != null; node = node.left) s1.push(node);

    TreeNode node2 = s2.pop();

    for(TreeNode node = node2.right; node != null; node = node.left) s2.push(node);

    }

    if(s1.empty() && s2.empty()) return true; else return false;

    }

    @Aahish: I think ur algo wont work when one binary Tree has repetition of elements while the other binary tree has a single instance of the repeated elements..

    I think slight adjustment will do .

    while traversing 1st binary tree hash each element of the tree along with its occurrence in a hash table.
    Now while traveling 2nd binary tree decrement the occurrence of each element found.
    After the completion of above steps , occurrence corresponding to each element in the hash table must be 0 .
    If not then the trees are not identical
