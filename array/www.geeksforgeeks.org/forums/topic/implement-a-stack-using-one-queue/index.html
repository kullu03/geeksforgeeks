    public String pop() {
    Iterator it = que.iterator();
    System.out.print(“Before Pop==>”);
    while (it.hasNext()) {
    System.out.print(it.next() + “\t”);
    }
    if (que.isEmpty())
    return null;
    else {
    que.add(myPop(que));
    que.add(reverse(que));
    return popped;
    }
    }

    private String myPop(Queue q) {
    String element = q.poll();
    if (q.isEmpty()) {
    popped = element;
    return null;
    } else {
    String temp = myPop(q);
    if (temp != null)
    q.add(temp);
    }
    return element;
    }

    private String reverse(Queue q) {
    String temp = q.poll();
    if (!q.isEmpty())
    q.add(reverse(q));
    return temp;
    }

    }
