    We can do something similar to http://www.geeksforgeeks.org/archives/19364 to solve this problem in O(n^3) time complexity. Following is complete code.


    
# include <stdio.h>
    

    
// A utility function to sort an array using Quicksort
    
void quickSort(int *, int, int);
    

    
// returns true if there are 4 numbers with sum equal
    
// to 'sum' present in A[]. Also, prints the 4 number
    
bool find4Numbers(int A[], int arr_size, int sum)
    
{
    
int l, r;
    

    
/* Sort the elements */
    
quickSort(A, 0, arr_size-1);
    

    
/* Now fix the first 2 elements one by one and find the
    
other two elements */
    
for (int i = 0; i < arr_size - 3; i++)
    
{
    
for (int j = i+1; j < arr_size - 2; j++)
    
{
    
l = j + 1; // index of the first element in the remaining elements
    
r = arr_size-1; // index of the last element
    
while (l < r)
    
{
    
if( A + A[j] + A[l] + A[r] == sum)
    
{
    
printf("%d, %d, %d, %d", A    , A[j], A[l], A[r]);
    
return true;
    
}
    
else if (A     + A[j] + A[l] + A[r] < sum)
    
l++;
    
else // A     + A[j] + A[l] + A[r] > sum
    
r--;
    
}
    
}
    
}
    
return false;
    
}
    

    
/* FOLLOWING 2 FUNCTIONS ARE ONLY FOR SORTING
    
PURPOSE */
    
void exchange(int *a, int *b)
    
{
    
int temp;
    
temp = *a;
    
*a   = *b;
    
*b   = temp;
    
}
    

    
int partition(int A[], int si, int ei)
    
{
    
int x = A[ei];
    
int i = (si - 1);
    
int j;
    

    
for (j = si; j <= ei - 1; j++)
    
{
    
if(A[j] <= x)
    
{
    
i++;
    
exchange(&A, &A[j]);
    
}
    
}
    
exchange (&A    , &A[ei]);
    
return (i + 1);
    
}
    

    
/* Implementation of Quick Sort
    
A[] --> Array to be sorted
    
si  --> Starting index
    
ei  --> Ending index
    
*/
    
void quickSort(int A[], int si, int ei)
    
{
    
int pi;    /* Partitioning index */
    
if(si < ei)
    
{
    
pi = partition(A, si, ei);
    
quickSort(A, si, pi - 1);
    
quickSort(A, pi + 1, ei);
    
}
    
}
    

    
/* Driver program to test above function */
    
int main()
    
{
    
int A[] = {1, 4, 45, 6, 10, 12};
    
int sum = 21;
    
int arr_size = sizeof(A)/sizeof(A[0]);
    

    
find4Numbers(A, arr_size, sum);
    

    
getchar();
    
return 0;
    
}

    @ kartik , have a look at below code. It runs in O(N^2) complexity.

    Pre-requisite: Sort the array before using the function:

    void findFourNumbers( int arr[], int size, int tofind )
    
{
    
int low, high, l, h, temp;
    

    
low = 0;
    
high = size - 1;
    

    
while( low < high )
    
{
    
temp = tofind - arr[low] - arr[high];
    

    
l = low + 1;
    
h = high - 1;
    

    
while( l < h )
    
{
    
if( arr[l] + arr[h] == temp )
    
{
    
printf("%d %d %d %dn",arr[low], arr[l], arr[h], arr[high] );
    
l++; h--; // find all such numbers
    
}
    
else if( arr[l] + arr[h] > temp )
    
h--;
    
else
    
l++;
    
}
    
if( arr[low] + arr[high] + arr[low+1] + arr[high-1] > tofind )
    
high--;
    
else
    
low++;
    
}
    
}
    

    
int main()
    
{
    
int arr[] = {1, 2, 3, 4, 5, 6, 7, 8},  tofind;
    

    
scanf("%d", &tofind);
    

    
findFourNumbers(arr, sizeof(arr)/sizeof(*arr), tofind);
    

    
return 0;
    
}
