    #include<iostream>
    
#include<string.h>
    
using namespace std;
    
int main()
    
{
    
char arr[]={"a1b2"};
    
int n=strlen(arr);
    
cout<<"Old array : "<<arr<<endl;
    
int i,j,k;
    
k=1;
    
while(k<(char)arr[n-1])
    
{
    
i=k;
    
j=n-1-k;
    
while(i<j)
    
{
    
swap(arr,arr[i+1]);
    
i+=2;
    
}
    
k++;
    
}
    
cout<<"New array : "<<arr<<endl;
    
return 0;
    
}

    Hi,

    It seems we can do it in O( (kn) ; where k is number of “char-int” pairs.

    Logic is simple : simply check whether any char comes after int then swap it that int.

    a1b2c3d4

    pass1 -> ab1c2d34 -> swapped 1-b, 2-c, 3-d

    pass2 -> abc1d234 -> swapped 1-c, 2-d

    pass3 -> abcd1234 -> swapped 1-d

    pass4 -> verify it’s all correct

    I assume that it has single digit int only.
