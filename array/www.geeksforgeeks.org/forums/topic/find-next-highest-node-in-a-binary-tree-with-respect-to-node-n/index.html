    {
    temp = temp->lChild;
    }
    else
    {
    temp = temp->rChild;
    }
    }
    if(temp) //node is present
    {
    if(val < final->info)

    }
    else

    final = final->rChild;
    }
    return final;
    }
    }

Node * getNextHighestNode(Node *root, int value) {
    if(root == NULL) return NULL;
    if(value >= root.data) return getNextHighestNode(root.right,value);
    Node *next = getNextHighestNode(root.left,value);
    return next == NULL ? root : next;
}
