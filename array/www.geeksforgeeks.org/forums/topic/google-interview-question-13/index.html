#include<stdio.h>
    
#include<stdlib.h>
    

    
/* A tree node structure */
    
struct node
    
{
    
int data;
    
struct node *left;
    
struct node *right;
    
};
    

    
/* Utility function to create a new Binary Tree node */
    
struct node* newNode (int data)
    
{
    
struct node *temp = new struct node;
    
temp->data = data;
    
temp->left = NULL;
    
temp->right = NULL;
    
return temp;
    
}
    

    
/* Given a binary tree, print its nodes in inorder*/
    
void printInorder1(struct node* node)
    
{
    
if (node == NULL)
    
return;
    

    
/* first recur on left child */
    
printInorder1(node->left);
    

    
/* then print the data of node */
    
printf("%d ", node->data);
    

    
/* now recur on right child */
    
printInorder1(node->right);
    
}
    

    
/* Given a binary tree, print its nodes in inorder*/
    
void printInorder2(struct node* node1, struct node *node2)
    
{
    
if (node1 == NULL)
    
{
    
printInorder1(node2);
    
return;
    
}
    
if (node2 == NULL)
    
{
    
printInorder1(node1);
    
return;
    
}
    
if (node1->data < node2->data)
    
{
    
printInorder2(node1, node2->left);
    
node2->left = NULL;
    
if (node1->right && node1->right->data > node2->data)
    
printInorder2(node1->right, node2);
    
else
    
printInorder1(node2);
    

    
}
    
else // node1->data > node2->data
    
{
    
printInorder2(node1->left, node2);
    
node1->left = NULL;
    
if (node2->right && node2->right->data > node1->data)
    
printInorder2(node1, node2->right);
    
else
    
printInorder1(node1);
    
}
    
}

{

    @karthik

    Basically i used normal iterative inorder traversal of tree …since we need to print the element in sorted form …so when ever i get smaller element from any of the tree i pop that from tree and print it otherwise push it back for the next iteration.

    Here the main reason for this code is it just takes O(log n) space and O(n) in case of balance tree …if you use recursion the implicit call stack will take O(n) space even in case of balance.

Node *get_next(Node *t, stack<Node *> &s) {
    
while (s.size() || t)
    
if (t) {
    
s.push(t);
    
t = t->left;
    
} else {
    
t = s.top();
    
s.pop();
    
return t;
    
}
    
return NULL;
    
}
