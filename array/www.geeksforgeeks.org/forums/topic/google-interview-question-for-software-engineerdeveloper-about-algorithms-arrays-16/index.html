    You can use modification of quick sort. 


    
int select(int * a,int start,int end,int n)
    
{
    
if (start == end)
    
return a[start]
    

    
int pivot = getPivot(a,i,j);//getpivot is an element which divides the array in two
    
// parts one part is greater than a[pivot] other is lower than a[pivot]
    
int mid = n/2;
    
if (pivot < mid)
    
return select(a,pivot,end,n);
    
else
    
if (pivot > mid)
    
return select(a,start,pivot,n)
    
else
    
return a[mid];
    
}

    Sort the first n+1 elements using quick sort. Since some one element occurs n times, this element occurs in these n+1 elements for at least two times.

    Perform a pairwise equivalence check as in bubble sort, the loop breaks when two adjacent elements are equal. And this element is the element that occurred n times.

# include<stdio.h>
    
# define bool int
    

    
int findCandidate(int *, int);
    
bool isMajority(int *, int, int);
    

    
/* Function to print Majority Element */
    
void printMajority(int a[], int size)
    
{
    
int last=a[size-1];
    
/* Find the candidate for Majority*/
    
int cand = findCandidate(a, size-1);
    

    
/* Print the candidate if it is Majority*/
    
if(isMajority(a, size-1, cand))
    
{ printf(" %d ", cand); return;}
    
else
    
{
    
int i=0,count=0;
    
int n=size/2;
    
for(i=0;i<size-1;i++)
    
{
    
if(a==last)
    
count++;
    
}
    
if(count==n-1)  //last element has count 1
    
printf("%d",last);
    
return;
    
}
    

    
printf("no such element");
    
}
    
/* Function to find the candidate for Majority */
    
int findCandidate(int a[], int size)
    
{
    
int maj_index = 0, count = 1;
    
int i;
    
for(i = 1; i < size; i++)
    
{
    
if(a[maj_index] == a    )
    
count++;
    
else
    
count--;
    
if(count == 0)
    
{
    
maj_index = i;
    
count = 1;
    
}
    
}
    
return a[maj_index];
    
}
    

    
/* Function to check if the candidate occurs more than n/2 times */
    
bool isMajority(int a[], int size, int cand)
    
{
    
int i, count = 0;
    
for (i = 0; i < size; i++)
    
if(a     == cand)
    
count++;
    
if (count > size/2)
    
return 1;
    
else
    
return 0;
    
}
    

    
/* Driver function to test above functions */
    
int main()
    
{
    
int a[] = {4, 1, 3, 4};
    
printMajority(a, 4);
    
getchar();
    
return 0;
    
}

    @guddu sharma:
    

    Logic “The idea is that two unique elements must be at most distance of 1.”
    

    It is not followed everytime. eg : 1 3 4 1 (distance between 1’s is more than 1 )
    

    

    One solution could be:
    

    1.  First assume A[0] is our desired value. Just traverse whole array and check its count.
    

    if (count==N) then we found the value. A[0] is the answer
    

    else
    

    2.  Consider Array B as =SubArray [1,2*n) {array except A[0] }
    

    Now problem is reduced to find Majority element in B. Which can be solved in O(n)
    

    To verify just check the count of majority element in whole array.
    

    Complexity: Max two traversal. O(n) time O(1) Space
