struct multilist
    
{

}*mlist;

}*llist;

3) add dummy to queue.
    
4) remove temp from queue
    
5) if(temp==dummy)
    
if(queue is not empty)
    
addtomultilist(llist,mlist);
    
llist=NULL;
    
add dummy to queue.
    
6)if temp != dummy
    
addtolist(temp->data,llist);
    
if temp has left child

if temp has right child
    
then add it to queue.
    
7) repeat from 4 till queue becomes empty.

    void link_sibling(struct node* root)
    {
    if(root == NULL) { return; }

    queue q;
    q.push(root);

    struct node* x = NULL;
    struct node* cur = NULL;

    while(!q.empty()) {
    x = q.back();

    do {
    cur = q.front();
    if(cur->left != NULL) q.push(cur->left);
    if(cur->right != NULL) q.push(cur->right);

    q.pop();
    if(cur != x)
    cur->next_sibling = q.front();
    } while(cur != x);
    }

    }
