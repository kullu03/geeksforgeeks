    Here is a solution using dynamic programming as described here.

    http://www.geeksforgeeks.org/dynamic-programming-subset-sum-problem

    After finding all the subarrays with the given sum, the code then reduces the result to those that are unique by first sorting each list using a hash table. The code is a bit long and complicated. Is there a better solution?

    public class AmazonSubsetSum {

    public static void main(String[] args) {
    dump(subsetSum(new int[] {2,2,3,1,2,1,1,1,1}, 4));

    }

    static class Sublist {
    public int[] arr;

    public Sublist(int[] arr) {
    this.arr = arr;
    }

    @Override
    public int hashCode() {
    int hash = arr.length;
    for (int a : arr) {
    hash ^= a;
    }
    return hash;
    }

    @Override
    public boolean equals(Object object) {
    Sublist list = (Sublist)object;
    return Arrays.equals(this.arr, list.arr);
    }
    }

    // Q(s,i) = is there a subset xi…xn with a sum of s?
    // Q(0,i>1) = true
    // Q(s,i) = Q(s,i-1) || Q(s-x,i-1)
    static HashSet<Sublist> subsetSum(int arr[], int sum) {
    boolean Q[][] = new boolean[sum+1][arr.length+1];

    // Q(0,i) = true
    for (int i = 0 ; i <= arr.length; ++i) {
    Q[0][i] = true;
    }

    for (int s = 1; s <= sum; ++s) {
    for (int i = 1; i <= arr.length; ++i) {
    Q[s][i] = Q[s][i-1];
    int subsum = s – arr[i-1];
    if (subsum >= 0) {
    Q[s][i] = Q[s][i] || Q[subsum][i-1];
    }
    }
    }

    HashSet<Sublist> result = new HashSet<Sublist>();
    Stack<Integer> stack = new Stack<Integer>();
    recurse(stack, Q, arr, sum, arr.length, result);

    return result;
    }

    static void recurse(Stack<Integer> stack, boolean Q[][], int arr[], int s, int i, HashSet<Sublist> result) {
    if (s == 0) {
    int subset[] = new int[stack.size()];
    for (int j = 0; j < stack.size(); ++j) {
    subset[j] = stack.get(j);
    }
    Arrays.sort(subset);
    result.add(new Sublist(subset));
    return;
    }
    if (Q[s][i-1]) {
    recurse(stack, Q, arr, s, i-1, result);
    }
    int subsum = s – arr[i-1];
    if (subsum >= 0 && Q[subsum][i]) {
    stack.push(arr[i-1]);
    recurse(stack, Q, arr, subsum, i-1, result);
    stack.pop();
    }
    }

    static void dump(HashSet<Sublist> result) {
    for (Sublist list : result) {
    for (int i : list.arr) {
    System.out.print(i + “,”);
    }
    System.out.print(“\n”);
    }
    }
    }

    template
    void swap( T arr[], int index, int i )
    {
    char temp = arr[index];

    arr[i]=temp;

    void permute( int a[], int index, int N)
    {
    vector v;
    for( int i=index; i v.push_back(a[i]);

    IntVector.push_back(v);

    int i=0;
    for( int i=index; i {
    swap( a, index, i );
    permute( a, index+1, N );
    swap( a, index, i );
    }
    }

    void PrintCombinationWithSum(int sum)
    {
    vector>::iterator it = IntVector.begin();
    while( it!=IntVector.end() )

    {
    curSum+=*iit;
    iit++;
