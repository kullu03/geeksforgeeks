    void MaxHeapify(int A[], int i)
    {
    int largest,temp,l,r;
    l=left(i);
    r=right(i);

    if(A[l]>A[i]&&l< =heapsize)
    largest=l;
    else
    largest=i;
    if(A[r]>A[largest]&&r< =heapsize)
    largest=r;

    if(largest!=i)
    {temp=A[i];
    A[i]=A[largest];
    A[largest]=temp;
    MaxHeapify(A,largest);
    }
    }
