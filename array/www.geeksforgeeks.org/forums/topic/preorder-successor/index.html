    Relatively simple one, since non-recursive preorder code is simple,


    

    
struct TreeNode {
    
TreeNode *lChild;
    
TreeNode *rChild;
    
int key;
    
};
    

    
typedef TreeNode *Tree;
    
Node *PreorderSuccessor(Tree root, TreeNode *pNode) {
    
if( !root || !pNode )
    
return NULL;
    

    
stack<TreeNode *> s;
    
TreeNode *pCrawl = root;
    

    
// Algorithm:
    
// Preorder traversal requires least amount of stack
    

    
// If the current node under process is pNode, three conditions can arise
    
// 1. If current node having left child, return this left child
    
// 2. If current node having right child, return this right child
    
// 3. If current node is a leaf node, return "right child of parent of parent of current node" (ideally it is on top of stack)
    

    
do {
    
if( pCrawl == pNode ) {
    
if( pCrawl->lChild )
    
return pCrawl->lChild;
    
else if( pCrawl->rChild )
    
return pCrawl->rChild;
    
else {
    
if( !s.empty() ) {
    
pCrawl = s.top();
    
return pCrawl;
    
}
    
else
    
return NULL;
    
}
    
}
    

    
if( pCrawl->rChild )
    
s.push(pCrawl->rChild);
    

    
if( pCrawl->lChild )
    
pCrawl = pCrawl->lChild;
    
else {
    
pCrawl = s.top();
    
s.pop();
    
}
    
} while( !s.empty() );
    

    
return NULL;
    
}

    if node has right child, then leftmost child in right subtree is preorder successor.

    else preorder successor is the parent in whose left subtree node lies

    code;


    
public Node PreorderSuccessor(Node root, Node n)
    
{
    
if (root == null)
    
{
    
return null;
    
}
    
Node pre = null;
    
if (n.right != null)
    
{
    
pre = n.right;
    
while (pre.left != null)
    
{
    
pre = pre.left;
    
}
    
}
    
else
    
{
    
Node p = root;
    
while (p != n)
    
{
    
if (n.data < p.data)
    
{
    
pre = p;
    
p = p.left;
    
}
    
else
    
{
    
p = p.right;
    
}
    
}
    
}
    
return pre;
    
}

    I think, we can work it through following algo:

    If the given node is not present simply return NULL;
    Now, if the given node is present:-
    Case 1:- It is a non-leaf node.
    ——–
    If its left child exists, then our preorder successor is this left child. But if left child is not present, then its right child is our answer.
    Case 2:- Its a leaf node
    ——-
    Take a variable of NOde type and each time store the right child if exist for each node you are traversing. If the right child id null, then dont update this value.
    Whenever you will reach the required leaf node, then this variable will be containing the preorder successor.

    For eg:
    ——
    A
    B C
    D E G
    Preorder sucessor of B is D(B is non leaf, so its left child)
    Preorder successor of C is G(C is non leaf, no left child , so right child is successor)
    For D, E is successor(variable will be having the value of E because iteratively when we came to B the variable val is updted with its right child.)

    Please correct me if you guys find anything wrong.

    // Inputs: root, preorder_of_node
    // Algo use property of BST to traverse
    // Uses rem variable to output the preorder successor if search element is leaf

    1. start
    2. current = root,rem = NULL;
    3. If (current == NULL) return NULL;
    4. If (preorder_of_node->element < current->element)
    4.a if (current->right != NULL) rem= current->right;
    4.b current = current->left;
    4.c goto step 3;
    5. else if (preorder_of_node->element > current->element)
    5.a current = current->right;
    5.b goto step 3;
    6. else if (preorder_of_node->element == current->element)
    6.a if (current->left != NULL) return current->left->element;
    6.b else if (current->right != NULL) return current->right->element;
    6.c else if (rem != NULL) return rem->element;
    6.d else return NULL;
    7. stop

struct node{
	int data;
	struct node* left;
	struct node* right;
};

void recinsert(struct node** p, int x)
{
	struct node * h,*r,*par;
	h = (struct node*)malloc(sizeof(struct node));
	h -> data = x;
	h->left = NULL;
	h->right = NULL;	
	if((*p) == NULL){
		(*p) = h;
		return;
	}	
	if((*p)->data <= x){
		recinsert(&(*p)->right,x);
	}else{
		recinsert(&(*p)->left,x);
	}
}

void recpreorder(struct node * p)
{
	if(p){
		cout<<p->data<<" ";
		recpreorder(p->left);
		recpreorder(p->right);
	}
}

	struct node * t;

				cout<<p->right->data<<endl;
			}else{
				while(!s.empty()){
					t = s.top();

		if(p->data < x){
			p = p->right;

		}

	}
}
