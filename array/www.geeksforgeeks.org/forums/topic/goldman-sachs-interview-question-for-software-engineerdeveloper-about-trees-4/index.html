    package tree;

    import java.util.ArrayList;
    import java.util.List;
    import java.util.Scanner;
    import java.util.Stack;

    class Tree{
    public Tree left;
    public Tree right;
    public int data;

    }

    public class LevelOrder {

    public static void main(String[] args) {
    Tree head = new Tree();

    System.out.println(“Enter the data of root node : “);
    Scanner scanIn = new Scanner(System.in);
    head.data = Integer.parseInt(scanIn.nextLine());

    inputTree(head,scanIn);

    displayTreeLevelOrder(head);

    distanceFromRoot(head,scanIn);

    distanceFromGiveNode(head,scanIn);

    scanIn.close();
    }

    private static void distanceFromGiveNode(Tree head, Scanner scanIn) {
    int first = -1, levelFirst = 0, levelSecond = 0, second = -1, level = 0;
    boolean findFirst = false, findSecond = false;

    System.out.println(“Enter first node “);
    first = Integer.parseInt(scanIn.nextLine());

    System.out.println(“Enter the node to be searched “);
    second = Integer.parseInt(scanIn.nextLine());

    List
    current = new ArrayList ();

    current.add(head);

    List next = new ArrayList ();

    while(true){
    for(Tree t : current){
    if(t.data == first){
    findFirst = true;
    levelFirst = level;
    }

    if(t.data == second){
    findSecond = true;
    levelSecond = level;
    break;
    }

    if(t.left != null){
    next.add(t.left);
    }

    if(t.right != null){
    next.add(t.right);
    }
    }

    if(next.size() == 0){
    break;
    }

    level++;
    current.clear();
    current.addAll(next);
    next.clear();
    }

    if(findFirst && findSecond){
    System.out.println(” Distance between the two nodes = “+(levelSecond – levelFirst));
    }
    else{
    System.out.println(“One of the nodes not found”);
    }
    }

    private static void distanceFromRoot(Tree head, Scanner scan) {
    int find = -1, level = 0;
    boolean result = false;

    System.out.println(“Enter the node to be searched “);

    find = Integer.parseInt(scan.nextLine());

    List current = new ArrayList ();

    current.add(head);

    List next = new ArrayList ();

    while(true){
    for(Tree t : current){
    if(t.data == find){
    result = true;
    break;
    }

    if(t.left != null){
    next.add(t.left);
    }

    if(t.right != null){
    next.add(t.right);
    }
    }

    if(next.size() == 0){
    break;
    }

    level++;
    current.clear();
    current.addAll(next);
    next.clear();
    }

    if(result){
    System.out.println(“Distance from root = “+level);
    }
    else{
    System.out.println(“Element not found”);
    }
    }

    private static void displayTreeLevelOrder(Tree head){
    List current = new ArrayList ();

    current.add(head);

    List next = new ArrayList ();

    System.out.println(“The tree is “);
    while(true){
    for(Tree t : current){
    System.out.print(t.data);

    if(t.left != null){
    next.add(t.left);

    if(t.right != null){
    next.add(t.right);
    }
    }

    System.out.println();

    private static void inputTree(Tree head, Scanner scanIn) {

    List current = new ArrayList ();
    current.add(head);

    List next = new ArrayList ();
    while(true){
    next.clear();

    for(Tree i: current){
    int left ,right;
    System.out.println(“Enter left node for “+i.data);
    left = Integer.parseInt(scanIn.nextLine());

    if(left != -1){
    Tree leftNode = new Tree();
    leftNode.data = left;
    i.left = leftNode;

    next.add(leftNode);
    }
    else{
    i.left = null;
    }

    System.out.println(“Enter right node for “+i.data);
    right = Integer.parseInt(scanIn.nextLine());

    if(right != -1){
    Tree rightNode = new Tree();
    rightNode.data = right;
    i.right = rightNode;

    next.add(rightNode);
    }
    else{
    i.right = null;
    }
    }

    if(next.size() == 0){
    break;
    }

    current.clear();
    current.addAll(next);
    }

    }

    }
