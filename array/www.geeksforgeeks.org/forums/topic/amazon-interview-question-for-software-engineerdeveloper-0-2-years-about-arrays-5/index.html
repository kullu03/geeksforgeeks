    It is a simple question of sorting and binary search. I don’t know why so much hassle about it.

    Let the given array be arr[]
    
1) Copy the given array to a temporary array temp[].
    
2) Sort the temp[] array using a O(nLogn) sorting algo
    
3) Traverse the given array arr[] from left to right.
    
a) For each element arri, binary search it in temp.
    
b) The index returned by binary search indicates the number of elements
    
smaller than arr    .  Store this index in ar_low.

    Use BST with one extra frequency field containing number of occurrences-1.

    Start from right to left, & for each item insert it into BST. If item is already there, increment its frequency.

    Whenever an item is found greater than the root while inserting, return 1 + freq of the root node.

    #include <stdio.h>
    

    
struct node
    
{
    
int data;
    
int freq;
    
struct node *left,*right;
    
};
    

    
struct node* getNode(int data)
    
{
    
struct node* temp=(struct node*)malloc(sizeof(struct node));
    
temp->left=temp->right=NULL;
    
temp->data=data;
    
temp->freq=0;
    
return temp;
    
}
    

    
int insert(struct node**root,int data)
    
{
    
if(!*root)
    
{
    
*root=getNode(data);
    
return 0;
    
}
    
else if(data<(*root)->data)
    
return insert(&((*root)->left),data);
    
else if(data>(*root)->data)
    
return 1 + (*root)->freq + insert(&((*root)->right),data);
    
else
    
{
    
++((*root)->freq);
    
return (*root)->freq;
    
}
    
}
    

    
void cal(int *arr,int size)
    
{
    
struct node *root=NULL;
    
int i,temp[size];
    
for(i=size-1;i>=0;i--)
    
{
    
temp=insert(&root,arr);
    
}
    
for(i=0;i<size;++i)
    
printf("%d ",temp    );
    
}
    

    
int main()
    
{
    
int arr[]={1,3,2,4,5,4,2},size;
    
size=sizeof(arr)/sizeof(arr[0]);
    
cal(arr,size);
    
return 0;
    
}

public class NewArray {

	 * You have an array like ar[]= {1,3,2,4,5,4,2}. You need to create
			another array ar_low[] such that ar_low = number of elements lower

	//method to find arr_ouput elements value
	public static int count(int [] arr_input, int num, int start) {
		int count =0;
		for(int i=start+1; i<arr_input.length;i++) {
			if(arr_input[i]<=num) {
				count++;
			}
		}
		return count;
	}

    It can be solved with BST in a much simpler manner as follows:
    Scan the elements from right to left and put them in a BST following the rules 

    int insert(Node* node, int value,int count) {
    if (value < node->key) {
        if (node->leftChild == NULL)
            node->leftChild = new Node(value);
        else
            insert(node->leftChild, value,count);
    } else {
        if(node->rightChild == NULL)
            node->rightChild = new Node(value);
        else{
            insert(node->rightChild, value);
            count++;
        }
    }
    return count;
}
