    Nice questions! I will try them all, but here are my initial thoughts on the second round:

     

    Problem 1 can be solved just as one would do it by hand:

    1. Move all numbers to the right side of the = sign and all x’s to the left

    2. If all the x’s cancel out, there is no solution.   (e.g. -5 + x = x = 5 because x can be any value)

    3. Sum the numbers and divide by coefficient of x

     

    Problem 2

    1. Use a greedy regexp pattern to find subproblems in parentheses.

    2. If the operator in front of the subproblem is a -, invert the signs of values in the subproblems

    3. Remove parentheses and repeat until there are no parentheses, then solve as problem 1.

     

    Problem 3

    This is a “pancake sort”. Sort from the bottom up. Every number can be moved into the correct position by flipping it to the top, then flipping it into the correct position. Once a number is in the correct position.

    }

     

     

    private static double solveForX(String equation) {

    // Removing whitespace between signs and numbers makes them easier to parse

    // Before: “x + 9 – 2 – 4 + x = -x + 5 – 1 + 3 – x”

    equation = equation.replaceAll(“\\-[\\s]*”, “-“);

    equation = equation.replaceAll(“\\+[\\s]*”, “+”);

    // After: “x +9 -2 -4 +x = -x +5 -1 +3 -x”

     

    // Create two arrays: one of the X values and one for the numbers

    ArrayList<String> Xs = new ArrayList<String>();

    ArrayList<Double> nums = new ArrayList<Double>();

     

    // Split the equation through the equals sign

    String[] sides = equation.split(“=”);

    String leftSide = sides[0];

    String rightSide = sides[1];

     

     

    // Parse the left side of the equation

    String[] leftValues = leftSide.split(” “);

    for(String value : leftValues) {

    if(value.length() == 0)

    continue;

    if(value.contains(“x”)) {

    Xs.add(value);

    } else {

    double dValue = Double.parseDouble(value);

    // numbers on the left sign need their sign inverted

    dValue = -dValue;

    nums.add(dValue);

    }

    }

     

    // Parse the right side of the equation

    String[] rightValues = rightSide.split(” “);

    for(String value : rightValues) {

    if(value.length() == 0)

    continue;

    if(value.contains(“x”)) {

    // X’s on the right sign need their sign inverted

                value = invertSignOfX(value);

                Xs.add(value);

    } else {

                double dValue = Double.parseDouble(value);

                nums.add(dValue);

    }

    }

     

    // Sum all the numbers

        double sum = 0;

        for(Double number : nums)

            sum += number;

     

        // Sum all the coefficients of X (which are either – or +)

        double coefficient = 0;

        for(String sign : Xs)

            if(sign.contains(“-“))

                coefficient–;

            else

                coefficient++;

     

            // X = sum of numbers / coefficient of X

            return sum / coefficient;

        }

     

     

        private static String invertSignOfX(String x) {

            if(x.contains(“-“))

                // Remove the -

                x = x.replaceAll(“\\-“, “”);

            else

                // Prepend a -

                x = “-” + x;

     

            return x;

        }

    }

    private static double solveForX(String equation) {

    // Removing whitespace between signs and numbers makes them easier to parse
    // Before: "x + 9 – 2 – 4 + x = -x + 5 – 1 + 3 – x"
    equation = equation.replaceAll("\\-[\\s]*", "-");
    equation = equation.replaceAll("\\+[\\s]*", "+");
    // After: "x +9 -2 -4 +x = -x +5 -1 +3 -x"

    // Split the equation through the equals sign
    String[] sides = equation.split("=");
    double sum = 0;
    double coefficient = 0;

    // Parse the left side of the equation
    double signSwap = 1;
    for(String side : sides) {
    String[] values = side.split(" ");
    for(String value : values) {
    if(value.length() == 0)
    continue;
    if(value.contains("x")) {
    coefficient += getSignOfX(value) * signSwap;
    } else {
    // numbers on the left sign need their sign inverted
    sum += Double.parseDouble(value) * -signSwap;
    }
    }
    signSwap *= -1;
    }

    // X = sum of numbers / coefficient of X
    return sum / coefficient;
    }

    private static int getSignOfX(String x) {
    if(x.contains("-"))
    return -1;

    return 1;
    }

    }

    typedef struct ComputeResult {
    double xCoeff;
    double sumOfConstants;
    } ComputeResult;

    double solveForX(const string& input) {
    size_t equals_pos = input.find(“=”);
    size_t length = input.length();
    double result = 0;

    if (string::npos == equals_pos)
    return -1;

    // Fill left computation result
    ComputeResult left;
    left.xCoeff = 0;
    left.sumOfConstants = 0;

    for (unsigned int i = 0; i < length; i++) {
    if (input[i] == '=')
    break;

    if (input[i] == '+' || input[i] == '-')
    continue;

    // LHS
    if (input[i] == 'x') {
    if (i > 0) {
    left.xCoeff = (input[i-1]==’-‘)?(left.xCoeff – 1):(left.xCoeff + 1);
    } else {
    left.xCoeff++;
    }
    } else {
    if (i > 0) {
    left.sumOfConstants = (input[i-1] == ‘-‘)?(left.sumOfConstants – (input[i] – ‘0’)):(left.sumOfConstants + (input[i] – ‘0’));
    } else {
    left.sumOfConstants = left.sumOfConstants + (input[i] – ‘0’);
    }
    }
    }

    // Fill right computation result
    ComputeResult right;
    right.xCoeff = 0;
    right.sumOfConstants = 0;

    // RHS
    for (unsigned int j = equals_pos+1; j < length; j++) {
    if (input[j] == '=') {
    return result;
    }

    if (input[j] == '+' || input[j] == '-')
    continue;

    if (input[j] == 'x') {
    if (j > 0) {
    right.xCoeff = (input[j-1]==’-‘)?(right.xCoeff – 1):(right.xCoeff + 1);
    } else {
    right.xCoeff++;
    }
    } else {
    if (j > 0) {
    right.sumOfConstants = (input[j-1] == ‘-‘)?(right.sumOfConstants – (input[j] – ‘0’)):(right.sumOfConstants + (input[j] – ‘0’));
    } else {
    right.sumOfConstants = right.sumOfConstants + (input[j] – ‘0’);
    }
    }
    }

    double xWeight = left.xCoeff – right.xCoeff;
    double constWeight = right.sumOfConstants – left.sumOfConstants;

    if ((xWeight != 0) && (constWeight != 0)) {
    result = constWeight/xWeight;
    }

    return result;
    }

    cout << solveForX(ip);

void swap(int *arr, size_t size, size_t i, size_t j)
{
    if (i == j) return;
    else if (i > j)
    { int k = i; i = j; j = k;}

//6 flip operations swap 2 elements

// iBjC    
   flip(a,0);
// C'jB'i    
   flip(a,n-j-1);
// C'iBj  
   flip(a,n-j);
// C'ijB'   
   flip(a,n-j+1);
// C'ijB  
   flip(a,n-j);

    `/* sort in O(nlogn) – essentially a modified quicksort */
    public void sort(int[] array) {
    if (array == null || array.length == 0) return;
    sort(array, 0, array.length – 1);
    }
    private void sort(int[] array, int left, int right) {
    // end case //
    if (left <= right) return;

    int mid = array[left + (right – left)/2];
    int pivot = partition(array, left, right, mid);
    sort(array, left, pivot – 1);
    sort(array, pivot + 1, right);
    }
    private void partition(int[] array, int left, int right, int index) {

    int pivot = array[index];
    while (left < right) {
    while (left < right && array[left] <= pivot) ++left;
    if (left == right) return left;
    while (right > left && array[right] > pivot) –right;
    if (left < right) swap(array, left, right);
    }
    return left;
    }
    private void swap(int[] array, int left, int right) {
    int max = array.length – 1;
    flip(array, left);
    flip(array, right);
    flip(array, left);
    }
