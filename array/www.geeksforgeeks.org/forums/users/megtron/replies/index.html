    I hope this would work fine.

    I just check for all words to form a grid in n^3 time. And then called checkGrid() to check whether the grid formed by the selected words has all the words in the dictionary. For the sake of brevity, I have just checked columns and diagonal in one direction (top to bottom).

    Please let me know if you find any bug.

    #include<iostream>

    #include<cstring>

    #include<cstdio>

    #include<cstdlib>

    #include<map>

    #define MAX 100

    using namespace std;

    map<string,int> mymap;

    bool checkGrid(string grid[])

    {

    int i,j;

    string temp;

    for(i=0;i<3;i++) //checking columns

    {

    temp.clear();

    for(j=0;j<3;j++)

    temp.push_back(grid[j]);

    if(mymap.find(temp)==mymap.end())

    return false;

    }

    //checking diagonal

    temp.clear();

    temp.push_back(grid[0][0]);temp.push_back(grid[1][1]);temp.push_back(grid[2][2]);

    if(mymap.find(temp)==mymap.end())

    return false;

    return true;

    }

    void selectWords(string s[],int n,string grid[])

    {

    for(int i=0;i<n;i++)

    {

    for(int j=0;j<n;j++)

    {

    if(j==i)

    continue;

    for(int k=0;k<n;k++)

    {

    if(k==i || k==j)

    continue;

    grid[0];

    grid[1][j];

    grid[2][k];

    //for(int l=0;l<3;l++)

    // cout<<grid[l]<<endl;

    if(checkGrid(grid))

    {

    cout<<“GRID formed using “<<i<<” “<<j<<” “<<k<<endl;

    for(int l=0;l<3;l++)

    cout<<grid[l]<<endl;

    cout<<endl;

    }

    }

    }

    }

    }

    void initialize(char* s[],char* grid[],int n)

    {

    s=new char[50];

    for(int i=0;i<3;i++)

    grid=new char[4];

    int main()

    string s[MAX];
