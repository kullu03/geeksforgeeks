			int level = a.findLevel(i);
			if(level>max){
				max = level;
			}
			a.level[i] = level;
		}

		int elementsDone = 0;
		while(elementsDone<N){
			for(int i =0;i<N;i++){
				if(a.level[i]==max){
					System.out.print(" "+i+" ");
					elementsDone++;
				}

			}
			max--;
			System.out.println();
		}

		
	}

	private int findLevel(int i) {
		if(arr[i]==-1)
			return 0;
		return (findLevel(arr[i])+1);
	}

    public int findBoundary(int low,int high, int N){
    int length = N;

    int mid = (high+low)/2;
    if((mid==0 && arr[mid]>arr[mid+1]) || (mid==length && arr[mid]>arr[mid-1]) || (mid!=0 && mid!=length && arr[mid]>arr[mid+1] && arr[mid]>arr[mid-1])){
    return mid;
    }
    else if(arr[mid+1]>arr[mid]){
    return findBoundary(low+1, high,N);
    }
    else{
    return findBoundary(low, high-1,N);
