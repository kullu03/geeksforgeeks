    #include<iostream>
    
using namespace std;
    
int numberOfStrings(int n, char A[], int has_b_occurred,int was_a_previous, int was_a_in_previous_two, char current_ch)
    
{
    
int count=0;
    
if(n==0)
    
{
    
A[current_ch]='';
    
cout<<A<<endl;
    
return 1;
    
}
    
else
    
{
    
A[current_ch]='c';
    
count += numberOfStrings(n-1,A,has_b_occurred,0,0,current_ch+1);
    

    
if(!has_b_occurred)
    
{
    
A[current_ch]='b';
    
count += numberOfStrings(n-1,A,1,0,0,current_ch+1);
    
}
    
if(!was_a_in_previous_two)
    
{
    
A[current_ch]='a';
    
if(was_a_previous)
    
{
    
count += numberOfStrings(n-1,A,has_b_occurred,was_a_previous,1,current_ch+1);
    
}
    
else
    
{
    
count += numberOfStrings(n-1,A,has_b_occurred,1,was_a_in_previous_two,current_ch+1);
    
}
    
}
    

    
return count;
    
}
    
}
    
int main()
    
{
    
int n;
    
cout<<"Enter size of array"<<endl;
    
cin>>n;
    
char A[n];
    
cout<<"Total number of strings is "<<numberOfStrings(n,A,0,0,0,0)<<endl;
    
}

j++;

#include <stdio.h>
    
#define M 3
    
#define N 4
    

    
int result[M][N];
    
int apples[M][N] ={
    
{1, 2, 5, 4},
    
{9, 7, 1, 2},
    
{8, 0, 1, 5}
    
};
    

    
int getMax(int i, int j)
    
{
    
if( (i<M) && (j<N) )
    
{
    
if( result[j] != 0 )
    
return result    [j];
    
else
    
{
    
int right = getMax(i, j+1);
    
int down = getMax(i+1, j);
    

    
result    [j] = ( (right>down) ? right : down )+apples[j];
    
return result    [j];
    
}
    
}
    
else
    
return 0;
    
}
    

    
int main()
    
{
    
printf("nnMax Apples = %dnn", getMax(0, 0));
    
}

int main()
    
{
    
int A[]={-2,-3,5,-6,8,-5,10,12,-16},max,sum=INT_MIN;
    
int n=sizeof(A)/sizeof(A[0]);
    
max=A[n-1];
    
for(int i=n-2;i>=0;i--)
    
{
    
if(A > max)
    
max=A    ;
    
else if(max-A     > sum)
    
sum=max-A    ;
    
}
    
cout<<"Max sum is: "<<sum<<endl;
    
return 0;
    
}

    Complexity: O(n)
