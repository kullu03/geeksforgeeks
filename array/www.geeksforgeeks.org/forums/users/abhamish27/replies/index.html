    levelOrderPrintReverse(Node root)

    {

    Stack<Node> S = new Stack();

    Queue<Node> Q = new ArrayDeque();

    Q.add(root);

    while (!Q.isEmpty())

    {

    Node n = Q.remove();

    S.push(n);

    if (n.right!=null)

    Q.add(n.right);

    if (n.left!=null)

    Q.add(n.left);

    }

    while(!S.empty())

    {

    Node n = S.pop();

    System.out.print(n.data);

    }

    }

    Instead you can start removing entries from the HashTable starting from -1

    which means that get the value for the key -1 from the hashtable and remove -1 and so on repeat till you are left with only

    one element. Please see above I have added the solutions.

    public static void findRoot(DistortedNode[] list){

    Map<Integer,Integer> buff = new HashMap<Integer, Integer>();

    int i=0;

    for(DistortedNode node :list){

             buff.put(node.index,i++);

    }

    int prevIndex= buff.get(-1);

    buff.remove(-1);

    while(buff.size()>0){

             int temp = buff.get(prevIndex);

             buff.remove(prevIndex);

             prevIndex =temp;

    }

    Another solution could be keep adding and removing while you are traversing the array –

    public static void findRoot_2(DistortedNode[] list){

    Set<Integer> buff = new HashSet<Integer>();

    for(int i=0;i<list.length;i++){

         int index = list[i].index;

         if(buff.contains(i)){

         buff.remove(i);

    }else{

         buff.add(i);

    }

    if(buff.contains(index)){

         buff.remove(index);

    }else{

    if(index!=-1)

       buff.add(index);

    }

    }

    for(Integer root:buff)   //Hey this is O(1) dive deep!  only one element will be left in the set

        printResult(list[root],root);

    }
