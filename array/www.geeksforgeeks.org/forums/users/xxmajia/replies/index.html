    public static int numOfDistinctBFS(int n) {
    if (n <= 1) {
    return 1;
    }

    int[] result = new int[n+1];
    result[0] = 1;
    result[1] = 1;
    for (int i = 2; i <= n; i++) {
    for (int left = 0; left < i; left++) {
    result[i] += result[left] * result[i-left-1];
    }
    }

    return result[n];
    }

    public static void doFindAll (int[] fibo, int target, int position, ArrayList result, ArrayList> results, int sum) {
    if (position == fibo.length) {
    if (sum == target) {
    results.add((ArrayList)result.clone());
    }
    } else {
    // case1: do not choose current num
    doFindAll(fibo, target, position+1, result, results, sum);

    // case2: choose current num if we can
    if (sum + fibo[position] == target) {
    result.add(fibo[position]);
    results.add((ArrayList)result.clone());
    result.remove(result.size()-1);
    } else if (sum + fibo[position] < target) {
    result.add(fibo[position]);
    doFindAll(fibo, target, position+1, result, results, sum+fibo[position]);
    result.remove(result.size()-1);
    }
    }
    }

    if (source == null || target == null || source.length() == 0 || target.length() == 0 || target.length() > source.length()) {
    return result;

    char[] s = source.toCharArray();
    char[] t = target.toCharArray();
    int[] map = new int[256];
    int count = 0;
    // init
    for (int i = 0; i < t.length; i++) {
    if (map[t[i]] == 0) {
    count++;
    }
    map[t[i]]++;
    }

    int start = 0;
    int end = 0;
    while (end < s.length) {
    if (map[s[end]] > 0) {
    start = end;
    int tempCount = count;
    int[] tempMap = new int[256];
    for (int i = 0; i < 256; i++) {
    tempMap[i] = map[i];
    }
    while (end < s.length && tempMap[s[end]] > 0) {
    tempMap[s[end]]–;
    if (tempMap[s[end]] == 0) {
    tempCount–;
    }
    end++;
    }

    if (tempCount == 0) {
    result.add(start);
    }
    } else {
    end++;
    }
    }

    return result;
    }

    Solution:
    run dfs on A[i][j] == 1, set cell to -1 if it can be reached from i,j.

    public static int maxArea(int[][] A) {
    if (A == null || A.length == 0 || A[0].length == 0) {
    return 0;
    }

    int max = 0;
    for (int i = 0; i < A.length; i++) {
    for (int j = 0; j < A[0].length; j++) {
    if (A[i][j] == 1) {
    max = Math.max(dfs(A, i, j), max);
    }
    }
    }

    return max;
    }

    public static class Coordinate {
    int i;
    int j;

    public Coordinate (int i, int j) {
    this.i = i;
    this.j = j;
    }
    }

    public static int dfs(int[][] A, int i, int j) {
    Stack stack = new Stack();
    A[i][j] = -1;
    int count = 1;
    stack.add(new Coordinate(i,j));
    while (!stack.isEmpty()) {
    Coordinate co = stack.pop();
    i = co.i;
    j = co.j;
    if (i-1 >= 0 && A[i-1][j] == 1) {
    A[i-1][j] = -1;
    stack.add(new Coordinate(i-1,j));
    count++;
    }

    if (i+1 < A.length && A[i+1][j] == 1) {
    A[i+1][j] = -1;
    stack.add(new Coordinate(i+1,j));
    count++;
    }

    if (j-1 >= 0 && A[i][j-1] == 1) {
    A[i][j-1] = -1;
    stack.add(new Coordinate(i,j-1));
    count++;
    }

    if (j+1 < A[0].length && A[i][j+1] == 1) {
    A[i][j+1] = -1;
    stack.add(new Coordinate(i,j+1));
    count++;
    }

    if (i-1 >= 0 && j-1 >= 0 && A[i-1][j-1] == 1) {
    A[i-1][j-1] = -1;
    stack.add(new Coordinate(i-1,j-1));
    count++;
    }

    if (i-1 >= 0 && j+1 < A[0].length && A[i-1][j+1] == 1) {
    A[i-1][j+1] = -1;
    stack.add(new Coordinate(i-1,j+1));
    count++;
    }

    if (i+1 < A.length && j+1 < A[0].length && A[i+1][j+1] == 1) {
    A[i+1][j+1] = -1;
    stack.add(new Coordinate(i+1,j+1));
    count++;
    }

    if (i+1 < A.length && j-1 >= 0 && A[i+1][j-1] == 1) {
    A[i+1][j-1] = -1;
    stack.add(new Coordinate(i+1,j-1));
    count++;
    }
    }

    return count;
    }
