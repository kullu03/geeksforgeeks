    if(lh>rh)
    return lh+1;

    else
    return rh+1;
    }

    node* recreverse(node *root,node **head){
    if(root==NULL)
    return NULL;

    recreverse(root->next,head);
    if(root->next==NULL){
    *head=root;
    return *head;
    }

    root->next->next=root;
    root->next=NULL;

    }
