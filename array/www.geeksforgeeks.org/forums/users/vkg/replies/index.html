    #include<iostream>
#include<string>
using namespace std;

void printPermutsUtil(string prefix, string s1, string s2, unsigned index1, unsigned index2)
{
    if(index1==s1.length())
    {
        cout<<prefix<<s2.substr(index2)<<endl;
        return;
    }
    if(index2==s2.length())
    {
        cout<<prefix<<s1.substr(index1)<<endl;
        return;
    }
    printPermutsUtil(prefix+s1[index1], s1, s2, index1+1, index2);
    printPermutsUtil(prefix+s2[index2], s1, s2, index1, index2+1);
}

void printPermuts(string s1, string s2)
{
    printPermutsUtil("", s1, s2, 0, 0);
}

int main(){
    string s1("AB"), s2("MNO");
    printPermuts(s1, s2);
    return 0;
}

    Simplified Pseudo-code for DP solution:

    for i=0 to size-1
	for j=0 to i-1
		if (an AP exists such that diff = a[i]-a[j] and ends at j )
			use this AP and update it to end at i
		else
			save info about new AP having 2 elems, ending at a[i] and diff = a[i]-a[j]
ans = non-existant AP with length = -1
for each AP
	if(AP.length > ans.length)
		ans = AP
return ans
