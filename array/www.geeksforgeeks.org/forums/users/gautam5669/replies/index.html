    Basically i used normal iterative inorder traversal of tree …since we need to print the element in sorted form …so when ever i get smaller element from any of the tree i pop that from tree and print it otherwise push it back for the next iteration.

    Here the main reason for this code is it just takes O(log n) space and O(n) in case of balance tree …if you use recursion the implicit call stack will take O(n) space even in case of balance.

void push(struct snode **s,struct node *k)
    
{
    
struct snode *tmp = (struct snode *) malloc(sizeof(struct snode));
    

    
//perform memory check here
    
tmp->t = k;
    
tmp->next = *s;
    
(*s) = tmp;
    

    
}
    

    
int isEmpty(struct snode *s)
    
{
    
if(s == NULL )
    
return 1;
    

    
return 0;

struct node *pop(struct snode **s)
    
{
    
struct  node *t;
    
struct snode *st;
    

    
st=*s;
    

    
(*s) = (*s)->next;
    

    
t = st->t;
    

    
free(st);
    

    
return t;
    
}

    Take the first element (i.e. a[0][0])

    insert into the min heap.

    keep a counter and call extract min

    and before extracting min add the it’s two neighbor elements in heap which has not been processed

    repeat this till u find the kth element.

    complexity (k logn)

    This gives a high complexity if K is order O(n^2)

    so we can always do the n^2 -k if and do same for Max heap if k is order of O(n^2)

# include<stdio.h>
    
# define bool int
    

    
int findCandidate(int *, int);
    
bool isMajority(int *, int, int);
    

    
/* Function to print Majority Element */
    
void printMajority(int a[], int size)
    
{
    
int last=a[size-1];
    
/* Find the candidate for Majority*/
    
int cand = findCandidate(a, size-1);
    

    
/* Print the candidate if it is Majority*/
    
if(isMajority(a, size-1, cand))
    
{ printf(" %d ", cand); return;}
    
else
    
{
    
int i=0,count=0;
    
int n=size/2;
    
for(i=0;i<size-1;i++)
    
{
    
if(a==last)
    
count++;
    
}
    
if(count==n-1)  //last element has count 1
    
printf("%d",last);
    
return;
    
}
    

    
printf("no such element");
    
}
    
/* Function to find the candidate for Majority */
    
int findCandidate(int a[], int size)
    
{
    
int maj_index = 0, count = 1;
    
int i;
    
for(i = 1; i < size; i++)
    
{
    
if(a[maj_index] == a    )
    
count++;
    
else
    
count--;
    
if(count == 0)
    
{
    
maj_index = i;
    
count = 1;
    
}
    
}
    
return a[maj_index];
    
}
    

    
/* Function to check if the candidate occurs more than n/2 times */
    
bool isMajority(int a[], int size, int cand)
    
{
    
int i, count = 0;
    
for (i = 0; i < size; i++)
    
if(a     == cand)
    
count++;
    
if (count > size/2)
    
return 1;
    
else
    
return 0;
    
}
    

    
/* Driver function to test above functions */
    
int main()
    
{
    
int a[] = {4, 1, 3, 4};
    
printMajority(a, 4);
    
getchar();
    
return 0;
    
}
