float get_avg_sum(std::vector< int >& coins, int S) {
    
std::vector< int > Q(S + 1, 100000); // min no of coins to get this sum
    
Q[0] = 0;
    
int i, s;
    
for(i = 0; i < coins.size(); i++)
    
for(s = 1; s <= S; s++) {
    
if(s >= coins)
    
Q = std::min(Q, Q[s - coins] + 1);
    
}
    
int avg = 0;
    
for(s = 0; s <= S; s++) {
    
avg += Q;
    
}
    
return ((float)avg/S);
    
}
    

    
std::vector< int > coins_set;
    
float min_avg = 1e10;
    
std::vector< int > opt_set;
    

    
void brute_force(int n_used, int coin_sum, int N, int S) {
    
float avg = get_avg_sum(coins_set, S);
    
if(min_avg > avg) {
    
min_avg = avg;
    
opt_set = coins_set;
    
}
    

    
if(n_used < N) {
    
int next = coins_set[n_used - 1] + 1;
    
for(; next <= S - coin_sum; next++) {
    
coins_set.push_back(next);
    
brute_force(n_used+1, coin_sum + next, N, S);
    
coins_set.pop_back();
    
}
    
}
    
}
    

    
int main() {
    
coins_set.push_back(1); // first coin is always 1
    
int N = 4;
    
brute_force(1, 1, N, 100);
    

    
printf("avg: %.3f; coins: [", min_avg);
    
for(int j = 0; j < opt_set.size(); j++) {
    
printf(" %d", opt_set[j]);
    
}
    
printf("]n");
    
return 1;
    
}

void transform_array2(char *s[], int len) {

    here is my solution, if anyone interested in this problem:

    we assign digits to characters from least significant to most significant in alternating way.

    That is, first we assign a digit to X then to Y, then again X[i-1] and Y[i-1] etc. up to X[0] and Y[0].

    each time we assgned a pair of digits, we check if the condition holds, i.e. X + Y = Z.

    If not, we do not need to explore this path anymore. The code is below:


    
int A[26]; // currently assigned digits
    
const char *s1 = "abcd",
    
*s2 = "dbcc",
    
*s3 = "bddx";
    

    
int l1, l2, l3;
    

    
// which = 0: assign the i-th character of string 's1'
    
// which = 1: assign the i-th character of string 's2'
    
// carry: carry flag for integer addition
    
void assign(int i, int which, int carry) {
    
// assign the i-th character of the 1st number
    
if(which == 0) {
    
if(i == -1) { // all combinations checked
    
if(carry == 1)
    
return;
    
for(int j = 0; j < l1; j++) {
    
printf("%d", A[s1[j] - 'a']);
    
}
    
printf(" + ");
    
for(int j = 0; j < l2; j++) {
    
printf("%d", A[s2[j] - 'a']);
    
}
    
printf(" = ");
    
for(int j = 0; j < l3; j++) {
    
printf("%d", A[s3[j] - 'a']);
    
}
    
printf("n");
    
return;
    
}
    
int i1 = s1 - 'a';
    
if(A[i1] != -1) {
    
assign(i, 1, carry); // assign i-th digit of 2nd no
    
return;
    
}
    
for(int d = 0; d < 10; d++) {
    
A[i1] = d;
    
assign(i, 1, carry);
    
}
    
A[i1] = -1;
    
return;
    
}
    
// assign the i-th character of the 2nd number
    
int x1 = A[s1     - 'a']; // s1 already assigned
    
int i2 = s2     - 'a';
    
int i3 = s3     - 'a';
    
if(A[i2] != -1) {
    
int sum = A[i2] + x1 + carry;
    
int c1 = sum / 10;
    
sum %= 10;
    
if(A[i3] != -1) {  // mismatch
    
if(A[i3] != sum)
    
return;
    
assign(i - 1, 0, c1);
    
return;
    
}
    
A[i3] = sum;
    
assign(i - 1, 0, c1);
    
A[i3] = -1;
    
return;
    
}
    
for(int d = 0; d < 10; d++) {
    
A[i2] = d;
    
int sum = A[i2] + x1 + carry;
    
int c1 = sum / 10;
    
sum %= 10;
    
if(A[i3] != -1) {  // mismatch
    
if(A[i3] != sum)
    
continue;
    
assign(i - 1, 0, c1);
    
} else {
    
A[i3] = sum;
    
assign(i - 1, 0, c1);
    
A[i3] = -1;
    
}
    
}
    
A[i2] = -1;
    
}
    

    
int main() {
    
memset(A, 0xff, sizeof(A));
    
l1 = strlen(s1);
    
l2 = strlen(s2);
    
l3 = strlen(s3);
    
assign(l1-1, 0, 0);
    
}
