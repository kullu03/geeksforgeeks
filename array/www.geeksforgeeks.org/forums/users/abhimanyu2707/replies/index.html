#include<stack>

using namespace std;

void display(stack<int> s)
{
	while(!s.empty())
	{
		cout<<s.top()<<"->";
		s.pop();
	}
}

void insertbottom(stack<int> &s, int a, int size)
{
	int num;
	if(size>0){
		num = s.top();
		s.pop();
		size--;
		insertbottom(s , a, size);
	}
	else
	{
		s.push(a);
		return;
	}
	s.push(num);
}

void reverse(stack<int> &s, int size)
{
	while(size > 0)
	{		
		int num = s.top();
		s.pop();
		size--;
		insertbottom(s, num, size);
	}
}

    Let the time taken be:

    Thu (Horse up)
    Thd (Horse down)
    Tfu (Foot up)
    Tfd (Foot down)

    Man starts with horse and come by foot :
    Thu + Tfd = t1 (t1 is known)

    Goes up with foot :
    Tfd + Tfu = t2

    Comes down on horse :
    Tfu + Thd = t3

    Goes up with horse :
    Thd + Thu = t4

    4 eq, 4 variables (as t1,t2,t3,t4 are known)
