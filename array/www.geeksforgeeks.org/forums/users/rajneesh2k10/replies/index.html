void display(FILE *fp, int N) {
    
char ch = fgetc(fp);
    
fseek(fp,-2,SEEK_CUR);
    
if (ch=='n' && N==1)
    
return;
    
if (ch=='n')
    
display(fp,N-1);
    
else
    
display(fp,N);
    
cout<<ch;
    
}

display(fp,atoi(argv[1]));
