    This problem can be solved using BST concept.
    
1> First of all i have created a node having five field.
    
struct Node{
    
char start;   //starting character of string
    
char end;   //ending character of string
    
bool visited;  //is node visited
    
struct Node *lc; //left child
    
struct Node *rc ; //right child
    
};
    
2>Then i created BST using these Node.
    
3> After that i checked one by one for each node that, is all node is forming cycle or not.
    
//My code-
    
#include <stdio.h>
    
#include <iostream>
    
using namespace std;
    
#include <stdlib.h>
    
#include <string.h>
    
#define row 6
    
#define col 10
    
int count=0;
    
struct BST{
    
char start;
    
char end;
    
bool visited;
    
struct BST *lc;
    
struct BST *rc;
    
};
    
struct BST* getNode(char start, char end)
    
{
    
struct BST *temp;
    
temp = (struct BST *)malloc(sizeof(struct BST));
    
temp->start = start;
    
temp->end = end;
    
temp->visited = false;
    
temp->lc = temp->rc =NULL;
    
return temp;
    
}
    
void createBST(struct BST **root, char start, char end)
    
{
    
struct BST *temp;
    
if(!(*root))
    
{
    
temp = getNode(start, end);
    
*root = temp;
    
}
    
else if( (*root)->start >= start)
    
createBST(&(*root)->lc, start, end);
    
else
    
createBST(&(*root)->rc, start, end);
    
}
    

    
int findChain(struct BST *root, char start, struct BST *root1)
    
{
    
if(!(root1))
    
return 0;
    
if(root1->start == start && !(root1->visited))
    
{
    
root1->visited = true;
    
count++;
    
//printf("%c %c %d n", root1->start, root1->end, count);
    
if(count == row)
    
return 1;
    
return findChain(root, root1->end, root);
    

    
}
    
if (root1->start >= start)
    
return findChain(root, start, root1->lc);
    
else if (root1->start < start)
    
return findChain(root, start,root1->rc);
    

    
}
    
void pushFalse(struct BST *root)
    
{
    
if(root)
    
{
    
root->visited = false;
    
pushFalse(root->lc);
    
pushFalse(root->rc);
    
}
    
}
    
int main()
    
{
    
//char arr[row][col]={"aaa","bbbb","ccc","dddd"};

char array[row][2];
    
int i;
    
struct BST *root=NULL;
    
for(i=0; i<row; i++)
    
{
    
char start = arr[0];
    
char end =  arr    [strlen(arr)-1];
    
array    [0] = start;
    
array    [1] = end;
    
createBST(&root, start, end);
    
}
    
//printf("preorder traversal of tree is n");
    
//preorder(root);

{
    
if(findChain(root, array    [0], root))
    
break;
    
pushFalse(root);
    
count=0;
    
}
    
if(count == row)
    
printf("chain found n");
    
else
    
printf("not foundn");
    

    
return 0;
    
}
