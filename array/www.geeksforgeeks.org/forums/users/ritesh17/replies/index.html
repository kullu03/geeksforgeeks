    struct node* createNode(int x)
    {
    struct node* newNode = (struct node*) calloc(1, sizeof(struct node));

    newNode->data = x;
    newNode->visited = false;
    newNode->left = newNode->right = NULL;

    return newNode;
    }

    int main(int argc, char const *argv[])
    {
    struct node* root;

    root = createNode(1);
    root->left = createNode(2);
    root->right = createNode(3);
    root->left->left = createNode(4);
    root->left->right = createNode(5);
    root->right->left = createNode(6);
    root->right->right = createNode(7);
    root->left->right->left = createNode(8);
    root->left->right->right = createNode(9);
    root->left->left->left = createNode(10);

    int t = 9;
    getAncestorsIteratively(root, t);

    return 0;
    }

    void getAncestorsIteratively(struct node* root, int target)
    {
    if(root == NULL) { return; }
    if(root->data == target) { cout < < "Target is root, Hence no ancestors.\n"; return; }

    stack s;
    bool l, r, leaf;

    s.push(root);
    while(!s.empty()) {

    while((s.top()->left != NULL) && (s.top()->left->visited == false)) {
    s.push(s.top()->left);
    s.top()->visited = true;
    if(s.top()->data == target) {
    printStack(s);
    return;
    }
    }

    while((s.top()->right != NULL) && (s.top()->right->visited == false)) {
    s.push(s.top()->right);
    s.top()->visited = true;
    if(s.top()->data == target) {
    printStack(s);
    return;
    }
    }

    // printStack(s);
    // cout < < s.top()->data < < " " << l << " " << r << " " << leaf << "\n";
    // cout << isLeaf(s.top()) << "\n";
    // return;

    while(true) {
    l = r = leaf = false;
    if(s.top()->left == NULL)
    l = true;
    else {
    if(s.top()->left->visited == true)
    l = true;
    else l = false;
    }

    if(s.top()->right == NULL)
    r = true;
    else {
    if(s.top()->right->visited == true)
    r = true;
    else r = false;
    }

    if(isLeaf(s.top()))
    leaf = true;

    // cout < < s.top()->data < < " " << l << " " << r << " " << leaf << "\n";

    if((l&&r) || leaf) {
    // cout << "Popped element: " << s.top()->data < < "\n";
    s.pop();
    // return;
    }
    else
    break;
    }

    }

    cout << "Element " << target << " not found in tree.\n"; // this part gives segfault

    // s.delete();
    }

    bool isLeaf(struct node* root)
    {
    if(root->left == NULL && root->right == NULL)
    return true;

    return false;

    int getMinCoinsForSum(int a[], int n, int sum);

    int main(int argc, char const *argv[])
    {
    int a[] = {1, 2, 3, 5};
    int n = sizeof(a)/sizeof(a[0]);
    int sum = 13;

    cout < < getMinCoinsForSum(a, n, sum);

    return 0;
    }

    int getMinCoinsForSum(int a[], int n, int sum)
    {
    int min[sum+1];
    int i, j, coins, rem, total;

    min[0] = 0;
    for(i=1; i<=sum; i++) {
    min[i] = INT_MAX;

    for(j=0; j if(i >= a[j]) {
    coins = i/a[j];
    rem = i%a[j];
    total = coins + min[rem];
    if(total < min[i])
    min[i] = total;
    }
    }
    }

    for(i=0; i<=sum; i++)
    cout << “sum: ” << i << ” requires min coins: ” << min[i] << “\n”;

    return min[sum];
    }

    void getProductComb(int a[], int n, int data[], int i, int size)
    {
    if(i == size) {
    show(data, size);
    return;
    }

    int j;
    for(j=n-size+i; j>=i; j–) {
    if(a[j] != data[i-1]) {
    data[i] = a[j];
    getProductComb(a, n, data, i+1, size);
    }
    }
    }

    {
    int i;

    {
    int m=3, n=5;

    // int a[] = {-1, 1, 2, 3, 4, 5, 1, 1, 1};

    getDuplicates(a, size);

    return 0;
    }

    void getDuplicates(int a[], int size)
    {
    int i, t, x;

    for(i=1; i if(a[i] == i)
    a[i] = -1;
    else {
    t = a[i];
    while(t>0) {
    x = a[t];
    if(a[t] > 0) a[t] = -1;
    else a[t] -= 1;
    t = x;
    }
    }
    }

    disp(a, size);

    for(i=1; i if(a[i] < -1)
    cout << i << ” “;
    cout << “\n”;
    }

    I tried but couldn’t derive a general formula which we can use to multiply 2 matrices represented by linked list.
    But I thought of a method which might work well.

    Lets say n = 2;
    Considering two 2*2 matrices for example,
    A = [1 2
    3 4]
    B = [5 6
    7 8]

    In their linked list forms we have –
    LA = 1->2->3->4
    LB = 5->6->7->8

    Consider two arrays,
    P is of size n*1
    Q is of size 1*n

    Construct P by breaking LA in intervals of n nodes.
    Means n nodes make 1 row of matrix P
    P = [1->2
    3->4]

    Similarly n nodes make 1 row of Q, but instead of connecting rows connect columns (this can be done in O(n))
    Q = [5 6
    | |
    7 8]

    Now write a generic function to multiply a column matrix (P) with a row matrix (Q)

    int sum(struct node* root)
    {
    int l, r;

    if(root == NULL) { return 0;}

    l = sum(root->left);
    r = sum(root->right);
    root->childsum = l+r+root->data; // field in struct node to hold child sum

    return root->childsum;
    }

    void link_sibling(struct node* root)
    {
    if(root == NULL) { return; }

    queue q;
    q.push(root);

    struct node* x = NULL;
    struct node* cur = NULL;

    while(!q.empty()) {
    x = q.back();

    do {
    cur = q.front();
    if(cur->left != NULL) q.push(cur->left);
    if(cur->right != NULL) q.push(cur->right);

    q.pop();
    if(cur != x)
    cur->next_sibling = q.front();
    } while(cur != x);
    }

    }
