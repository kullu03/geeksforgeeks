    <?php
    

    
$s = [1, 2, 3, 4, 5];
    
$k = 7;
    

    
$sets  = [];
    
$count = count($s);
    
$total = pow(2, $count); // 2^5 = 32
    

    
for ($i = 0; $i < $total; $i++) {
    
$binary = sprintf('%0' . $count . 'b', $i); // 00000, 00001, 00010, ...
    

    
$set = [];
    

    
for ($j = 0; $j < $count; $j++) {
    
if ($binary[$j] === '1') {
    
$set[] = $s[$j];
    
}
    
}
    

    
$sets[] = $set;
    
}
    

    
$sets = array_filter($sets, function($set) use ($k) {
    
return array_sum($set) <= $k;
    
});
    

    
$sets = array_filter($sets, function($set) use ($sets) {
    
foreach ($sets as $tmp) {
    
if ($set === $tmp) {
    
continue;
    
}
    

    
$diff = array_diff($set, $tmp);
    

    
if (empty($diff)) {
    
return false;
    
}
    
}
    

    
return true;
    
});

    <?php
    

    
$input = file('input.txt', FILE_IGNORE_NEW_LINES);
    
$t     = array_shift($input);
    

    
foreach ($input as $k => $line) {
    
list($w, $h, $text) = explode(' ', $line, 3);
    

    
for ($s = 1000; $s > 0; $s--) {
    
$width = floor($w / $s);
    

    
if ($width == 0) continue;
    

    
$billboard = wordwrap($text, $width);
    
$proof = wordwrap($text, $width, PHP_EOL, true);
    

    
if ($billboard !== $proof) continue;
    

    
$lines = explode(PHP_EOL, $billboard);
    

    
if (count($lines) * $s <= $h) break;
    
}
    

    
printf('Case #%d: %d' . PHP_EOL, $k+1, $s);
    
}
