#include<iostream>
    
#include<stdio.h>
    
using namespace std;
    

    
void swap(int *x, int *y) {
    
int temp = *x;
    
*x = *y;
    
*y = temp;
    
}
    

    
class Heap
    
{
    
int *arr; // pointer to array of elements in heap
    
int heap_size;
    
public:
    
Heap(int a[], int size);
    
void minHeapify(int );
    
int getRoot() {return arr[0];}
    
int parent(int i){ return (i-1)/2; } ;
    
int left(int i)  { return (2*i + 1); } ;
    
int right(int i) { return (2*i + 2); } ;
    
};
    

    
Heap::Heap(int a[], int size) {
    
heap_size = size;
    
arr = a;
    
}
    

    
void Heap::minHeapify(int i) {
    
int l = left(i);
    
int r = right(i);
    
int largest;
    
if (l < heap_size && arr[l] < arr)
    
largest = l;
    
else
    
largest = i;
    
if (r < heap_size && arr[r] < arr[largest])
    
largest = r;
    
if (largest != i)
    
{
    
swap(&arr    , &arr[largest]);
    
minHeapify(largest);
    
}
    
}
