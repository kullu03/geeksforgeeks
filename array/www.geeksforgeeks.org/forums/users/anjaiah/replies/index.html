    public void printLastnNodes(LLNode head, int n) {

    //in java

    LLNode first = head, second = head;

    // increment the first pointer by n nodes.

    int count = 0;

    while (count <= n && first != null) {

    first = first.next;

    count++;

    }

    if (first == null) {

    // if list size is less than n then print all.

    while (second != null) {

    System.out.print(second.data + “->”);

    second = second.next;

    }

    } else {

    // if list size is more than n then, place second pointer at n th node and start printing.

    while (first != null) {

    first = first.next;

    second = second.next;

    }

    //at this point first is pointing to last node, so second is pointing to n-1 node.

    second = second.next;//now second points to nth node.

    while (second != null) {

    System.out.print(second.data + “->”);

    second = second.next;

    }

    }

    }

    public LLNode sortLL_buble(LLNode head) {

    // Bubble sort using java

    int temp;

    for (LLNode i = head; i != null; i = i.next) {

    for (LLNode j = head; j.next != null; j = j.next) {

    if (j.data > j.next.data) {

    temp = j.data;

    j.data = j.next.data;

    j.next.data = temp;

    }

    }

    }

    return head;

    }

    int myAdd(int a, int b) {

    int and = a & b;
