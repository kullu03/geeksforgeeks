    Static allocation;

    int aArray[ROW][COL];

    Dynamic Allocation:

    int **aArray;

    aArray=(int **)(malloc(sizeof(int)*ROW));

    aArray[0]=(int*)(malloc(sizeof(int)*COL*ROW));

    for(i=0;i<ROW;i++)

    aArray=aArray[0]+i*COL;

    there are many ways to allocate dynamically i am showing the one of them.

    If you want to do this in minimum no. of malloc call its here

    int ** Dynamic2D(int **aArray, int ROW,int COL){

    int header=ROW*sizeof(int*);

    int data=ROW*COL*sizeof(int);

    aArray=malloc(data+header);

    int *npBuff=aArray+ROW;

    for(i=0;i<COL;i++)

    aArray=npBuff+i*COL;

    return aArray;

    }
