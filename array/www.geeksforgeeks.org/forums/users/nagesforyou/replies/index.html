    package nonlinear;

    class BSTNode {

    char data;

    BSTNode left;

    BSTNode right;

    BSTNode(char data) {

    this.data = data;

    }

    public char getData() {

    return this.data;

    }

    public void setRight(BSTNode right) {

    this.right = right;

    }

    public BSTNode getRight() {

    return this.right;

    }

    public void setLeft(BSTNode left) {

    this.left = left;

    }

    public BSTNode getLeft() {

    return this.left;

    }

    public String toString() {

    StringBuffer buffer = new StringBuffer();

    buffer.append(“[ “);

    if (this.left != null)

    buffer.append(left.toString());

    buffer.append(” ” + this.data + ” “);

    if (this.right != null)

    buffer.append(right.toString());

    buffer.append(” ]”);

    return buffer.toString();

    }

    }

    public class BSTNL {

    BSTNode root;

    int pos;

    public void print() {

    System.out.println(root);

    }

    public void createTree(String str) {

    char[] c = str.toCharArray();

    pos = 0;

    root = create(c);

    }

    private BSTNode create(char[] c) {

    if (pos == c.length)

    return null;

    if (c[pos++] == ‘N’) {

    BSTNode temp = new BSTNode(‘N’);

    temp.setLeft(create(c));

    temp.setRight(create(c));

    return temp;

    } else {

    BSTNode temp = new BSTNode(‘L’);

    return temp;

    }

    }

    public static void main(String[] args) {

    BSTNL t = new BSTNL();

    t.createTree(“NNNLLLNNLLL”);

    t.print();

    }

    }

    package array;

    public class FindSumOfK {

    public static void main(String[] args) {

    int[] arr = new int[] { 0, 2, 4, 6, 7, 8, 12, 14, 16 };

    find(arr, 16);

    }

    public static void find(int[] arr, int value) {

    int minPos = arr.length – 1;

    for (int i = 0; i <= minPos; ++i) {

    int expected = value – arr;

    int pos = search(arr, i, arr.length – 1, expected);

    if (pos >= 0 && pos != i) {

    System.out.println(arr + ” ” + expected);

    minPos = pos;

    }

    }

    }

    private static int search(int[] arr, int start, int end, int expected) {

    int mid = start + (end – start) / 2;

    if (arr[mid] == expected)

    return mid;

    if (start == end)

    return -1;

    else if (start + 1 == end)

    return arr[end] == expected ? end : -1;

    else if (arr[mid] < expected)

    return search(arr, mid, end, expected);

    else

    return search(arr, start, mid, expected);

    }

    }
