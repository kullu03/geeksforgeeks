    Assuming, the first part is sorted in increasing order and the second part in decreasing order. The below program can be modified easily to work for the other case.

    int findSteep(int arr[], int l, int h) {

    if (l > h)        return -1;

        if (l == h)
    return l;

    if (l == h – 1)
    return arr[l] < arr[h] ? l : h;

    int mid = (l + h) >> 1;

    if (arr[mid] < arr[mid – 1] && arr[mid] > arr[mid + 1])
    return mid;

    if (arr[mid] > arr[l])
    return findSteep(arr, mid, h);
    return findSteep(arr, l, mid – 1);
    }

  • September 12, 2012 at 8:41 PM in reply to: lvalue and rvalue doubt #113275

    Aashish
    Moderator

    I will try to explain the answer with an example:

    Consider the below code snippet written in C:

    int i;
    
int* ptr = &i;

    Here “i” is l-value, while &i is r-value. “&” used is “address-of” operator.

    A little modification in the above code snippet, and written in C++:

    int i;
    
int &ptr = i;

    Here, there is a wrong impression that &ptr is a r-value. The “&” operator used here is not “address-of” operator. The “&” is just the syntax indicating that ptr is a reference-to-int, it’s not taking the address of ptr.

    Follow the below link for better understanding:

    http://stackoverflow.com/questions/8488606/will-an-lvalue-to-rvalue-conversion-happen

    Use sliding window approach.

    I am assuming that the longest substring would not be containing duplicate characters.

    e.g. owacbajkl, the length of substring “acba” is 4.

    So, it must contain characters spread over an ascii value p to p+4 where p is the minimum ascii value in the substring.

    void fillTable(char* str,int j,int k,int* Table)
    
{
    
int i;
    
for(i=j;i<=k;++i)
    
Table[str-97]++;
    
}
    

    
int isContiguous(int* Table,int size)
    
{
    
int count=0,i,foundFirstOccurrence=0;
    
for(i=0;i<26;++i)
    
{
    
if(!foundFirstOccurrence && Table    )
    
{
    
foundFirstOccurrence=1;
    
count=1;
    
}
    
else if(foundFirstOccurrence && !Table    ) break;
    
else if(Table    )
    
++count;
    
}
    
return count==size;
    
}
    

    
void findMaxSize(char* str)
    
{
    
int n=strlen(str),gap,j,k,maxSize,start;
    
int Table[26];
    

    
for(gap=1;gap<n;++gap)
    
{
    
for(j=0,k=gap;k<n;++k,++j)
    
{
    
memset(Table,0,26);
    
fillTable(str,j,k,Table);
    
if(isContiguous(Table,gap+1))
    
{
    
maxSize=gap;
    
start=j;
    
}
    
}
    
}
    
for(j=start;j<=start+maxSize;++j)
    
printf("%c",str[j]);
    
}

    There are multiple problems in your code:

    1. You have declared the A class constructor private. So, object of class B can’t be created.

    2. In order to handle the exceptions, three keywords are must. try, throw & catch. The exception handler will catch the exceptions only if you throw one.

    Here is the modified code that works fine.

    class A
    
{
    
public:
    
A()
    
{
    
cout << "creating A" << endl;
    
try
    
{
    
int a=1,b=0;
    
if(b==0)
    
throw b;
    
a=a/b;
    
}
    
catch (...)
    
{
    
cout << "exception occured"<< endl;
    
}
    
}
    

    
~A()
    
{
    
cout << "deleting A" << endl;
    
}
    
};
    

    
class B : A
    
{
    
public:
    
B()
    
{
    
cout << "creating B" << endl;
    
}
    

    
~B()
    
{
    
cout << "deleting B" << endl;
    
}
    
};
    

    
int main()
    
{
    
B b;
    
return 0;
    
}
