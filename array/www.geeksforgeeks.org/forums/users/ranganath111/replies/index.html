    int[] a = {1,3,4,7,8,10,13,15,21};
    for(int x = a.length -1; x>=0;x–)
    {
    int i = 0;
    int j = x-1;
    while(i<j)
    {
    if(a[i] + a[j] > a[x])
    {
    j–;
    }
    else if(a[i] + a[j] < a[x])
    {
    i++;
    }
    else
    {
    System.out.println(“Pair:” + a[i] +” + ” + a[j] + ” = ” + a[x]);
    break;
    }
    }
    }
