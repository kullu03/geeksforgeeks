Maximum circular subarray sum

Given n numbers (both +ve and -ve), arranged in a circle, fnd the maximum sum of consecutive number.

Examples: 

Input: a[] = {8, -8, 9, -9, 10, -11, 12}
Output: 22 (12 + 8 - 8 + 9 - 9 + 10)

Input: a[] = {10, -3, -4, 7, 6, 5, -4, -1} 
Output:  23 (7 + 6 + 5 - 4 -1 + 10) 

Input: a[] = {-1, 40, -14, 7, 6, 5, -4, -1}
Output: 52 (7 + 6 + 5 - 4 - 1 - 1 + 40)

There can be two cases for the maximum sum:

Case 1: The elements that contribute to the maximum sum are arranged such that no wrapping is there. Examples: {-10, 2, -1, 5}, {-2, 4, -1, 4, -1}. In this case, Kadane’s algorithm will produce the result.

Case 2: The elements which contribute to the maximum sum are arranged such that wrapping is there. Examples: {10, -12, 11}, {12, -5, 4, -8, 11}. In this case, we change wrapping to non-wrapping. Let us see how. Wrapping of contributing elements implies non wrapping of non contributing elements, so find out the sum of non contributing elements and subtract this sum from the total sum. To find out the sum of non contributing, invert sign of each element and then run Kadane’s algorithm.
Our array is like a ring and we have to eliminate the maximum continuous negative that implies maximum continuous positive in the inverted arrays.

Finally we compare the sum obtained by both cases, and return the maximum of the two sums.

Thanks to ashishdey0 for suggesting this solution. Following is C implementation of the above method.

// Program for maximum contiguous circular sum problem
#include<stdio.h>

// Standard Kadane's algorithm to find maximum subarray sum
int kadane (int a[], int n);

// The function returns maximum circular contiguous sum in a[]
int maxCircularSum (int a[], int n)
{
   // Case 1: get the maximum sum using standard kadane's algorithm
   int max_kadane = kadane(a, n);

   // Case 2: Now find the maximum sum that includes corner elements.
   int max_wrap  =  0, i;
   for(i=0; i<n; i++)
   {
        max_wrap += a[i]; // Calculate array-sum
        a[i] = -a[i];  // invert the array (change sign)
   }

   // max sum with corner elements will be:
   // array-sum - (-max subarray sum of inverted array)
   max_wrap = max_wrap + kadane(a, n);

   // The maximum circular sum will be maximum of two sums
   return (max_wrap > max_kadane)? max_wrap: max_kadane;
}

// Standard Kadane's algorithm to find maximum subarray sum
// See http://www.geeksforgeeks.org/archives/576 for details
int kadane (int a[], int n)
{
    int max_so_far = 0, max_ending_here = 0;
    int i;
    for(i = 0; i < n; i++)
    {
        max_ending_here = max_ending_here + a[i];
        if(max_ending_here < 0)
            max_ending_here = 0;
        if(max_so_far < max_ending_here)
            max_so_far = max_ending_here;
    }
    return max_so_far;
}

/* Driver program to test maxCircularSum() */
int main()
{
    int a[] =  {11, 10, -20, 5, -3, -5, 8, -13, 10};
    int n = sizeof(a)/sizeof(a[0]);
    printf("Maximum circular sum is %d\n", maxCircularSum(a, n));
    return 0;
}

Output: 

Maximum circular sum is 31

Time Complexity: O(n) where n is the number of elements in input array.

Note that the above algorithm doesn’t work if all numbers are negative e.g., {-1, -2, -3}. It returns 0 in this case. This case can be handled by adding a pre-check to see if all the numbers are negative before running the above algorithm.

    A more efficient approach. Space complexity O(n):

     
#include<stdio.h>
#include<stdlib.h>

int maxCircularSum(int array[], int n) {
  int current_index;
  int sum=0, max_sum=0, subarray_start=-1;

  /* Create an array of double size than the given array and fill it twice with the elements as below: */
  /* repeat_elements[i]=repeat_elements[n+i]=array[i] */ 
  int * repeat_elements=(int *)malloc(sizeof(int)*n*2);
  for(current_index=0; current_index<n; current_index++) {
    *(repeat_elements+current_index)=array[current_index];
    *(repeat_elements+n+current_index)=array[current_index];
  }

  /* subarray_start has the index of the start of subarray */
  for(current_index=0; current_index<2*n-1;current_index++) {
    /* If next element is taking the length of subarray more than n that means the end elements */
    /* are repeating and so you need to move subarray_start forward and ignore the leftmost */
    /* element in the subarray. If at any point sum becomes more than max_sum update the value */
    /* of max_sum accordingly. Next the while loop is to remove all the negative values from the start. */ 
    if(subarray_start>=0 && current_index-subarray_start+1>n) {
      sum=sum-repeat_elements[subarray_start];
      subarray_start=subarray_start+1;
      if(sum>max_sum) {
        max_sum=sum;
      }
      while(subarray_start>=0 && repeat_elements[subarray_start]<0 && subarray_start<current_index-1) {
        sum=sum-repeat_elements[subarray_start];
        subarray_start=subarray_start+1;
        if(sum>max_sum) {
          max_sum=sum;
        }
      }
    }

    /* If sum became zero in above step start a fresh and mark current_index as subarray_start. */
    if(sum==0) {
      subarray_start=current_index;
    }
    /* Add value at current_index in the sum. */
    sum=sum+repeat_elements[current_index]; 

    /* If sum becomes negative again start a fresh and mark current_index+1 the new subarray_start. */
    if(sum<0) {
      sum=0;
      subarray_start=current_index+1;
    }
    /* If sum is greater than max_sum then update max_sum */
    if(sum>max_sum) {
      max_sum=sum;
    }

    /* If subarray_start moves beyond index n-1 that means you shall get the same results as you had in */
    /* the first half of the array and so just return max_sum */
    if(subarray_start>=n) {
      return max_sum;
    }
  }
  return max_sum;
}

int main()
{
    int array[] =  {-1, 40, -14, 7, 6, 5, -4, -1};
    int n = sizeof(array)/sizeof(array[0]);
    printf("Maximum circular sum is %d\n", maxCircularSum(array, n));
    return 0;
}

#include<stdio.h>

int maxCircularSum(int array[], int n) {
  int i, j, k;
  int sum;
  int max_sum=0;
  int max_sum_array_start=-1;
  int max_sum_array_end=-1;

  for(i=0; i<n; i++) {
    for(j=0; j<n; j++) {
      sum=0;
      if(i==j) {
        sum=array[i];
      } else if(i<j) {
        for(k=i; k<=j; k++) {
          sum=sum+array[k];
        }
      } else if(i>j) {
        for(k=i; k<n; k++) {
          sum=sum+array[k];
        }
        for(k=0; k<=j; k++) {
          sum=sum+array[k];
        }
      }
      if(sum>max_sum) {
        max_sum=sum;
        max_sum_array_start=i;
        max_sum_array_end=j;
      }
    }
  }
  printf("Subarray between array indexes %d and %d has the maximum sum %d\n", max_sum_array_start, max_sum_array_end, max_sum);
  return max_sum;
}

    Max subarray from circular array can be find only by running kadane algo once.

    Solution is very simple.

    1. Find the index of most negative number in array lets say negIndex . ( If all number are positive then max sum is total sum of array )

    2. Find maxSubArray from negIndex+1 to negIndex-1

    Eg. 10, -3, -4, 7, 6, 5, -4, -1

    Here most negative index is 2 i.e -4
    MaxSum ( 7 6 5 -4 -1 10 -3 ) is the answer
    which is 23

    Proof is simple : In circular array biggest negative number will always contribute in decreasing the sum.

    Edge Case : Check if all number are negative then pick smallest negative number as answer.

// Program for maximum contiguous circular sum problem
#include<stdio.h>

// Standard Kadane's algorithm to find maximum subarray sum
int kadane (int a[], int n);

// The function returns maximum circular contiguous sum in a[]
int maxCircularSum (int a[], int n) {
   // Case 1: get the maximum sum using standard kadane's algorithm
   int max_kadane = kadane(a, n), all_neg = 1;

   // Case 2: Now find the maximum sum that includes corner elements.
   int max_wrap  =  0, i;
   for(i=0; i<n; i++) {
     if ( a[i] >= 0 )
       all_neg = 0 ;
        max_wrap += a[i]; // Calculate array-sum
        a[i] = -a[i];  // invert the array (change sign)
   }

   if ( all_neg )
      return max_kadane ;
   // max sum with corner elements will be:
   // array-sum - (-max subarray sum of inverted array)
   max_wrap = max_wrap + kadane(a, n);

   // The maximum circular sum will be maximum of two sums
   return (max_wrap > max_kadane)? max_wrap: max_kadane;
}

// Standard Kadane's algorithm to find maximum subarray sum
// See http://www.geeksforgeeks.org/archives/576 for details
int kadane (int a[], int n)
{
    int max_so_far = a[0], max_ending_here = 0;
    int i;
    for(i = 0; i < n; i++)
    {
        max_ending_here = max_ending_here + a[i];
        if(max_so_far < max_ending_here)
            max_so_far = max_ending_here;
        if(max_ending_here < 0)
            max_ending_here = 0;
    }
    return max_so_far;
}
